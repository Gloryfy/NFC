103 status_t LPSPI_DRV_SlaveInit(uint32_t instance,
104  lpspi_state_t * lpspiState,
105  const lpspi_slave_config_t * slaveConfig)
106 {
107  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
108  DEV_ASSERT(lpspiState != NULL);
109  DEV_ASSERT(slaveConfig != NULL);
110  LPSPI_Type * base = g_lpspiBase[instance];
111  status_t errorCode = STATUS_SUCCESS;
112 
113  lpspiState->lsb = slaveConfig->lsbFirst;
114  lpspiState->bitsPerFrame = slaveConfig->bitcount;
115  lpspiState->transferType = slaveConfig->transferType;
116  lpspiState->isBlocking = false;
117  /* Store DMA channels numbers used for DMA transfer */
118  lpspiState->rxDMAChannel = slaveConfig->rxDMAChannel;
119  lpspiState->txDMAChannel = slaveConfig->txDMAChannel;
120  /* Store callback */
121  lpspiState->callback = slaveConfig->callback;
122  lpspiState->callbackParam = slaveConfig->callbackParam;
123  /* Calculate the bytes/frame for lpspiState->bytesPerFrame. */
124  lpspiState->bytesPerFrame = (uint16_t)((lpspiState->bitsPerFrame + 7U) / 8U);
125  /* For DMA transfers bytes per frame must be equal to 1, 2 or multiple of 4. */
126  if ((lpspiState->transferType == LPSPI_USING_DMA) && (!(((lpspiState->bytesPerFrame % 4U) == 0U) ||
127  (lpspiState->bytesPerFrame <= 2U))))
128  {
129  return STATUS_ERROR;
130  }
131  lpspiState->isTransferInProgress = false;
132  /* Initialize the semaphore */
133  errorCode = OSIF_SemaCreate(&(lpspiState->lpspiSemaphore), 0);
134  DEV_ASSERT(errorCode == STATUS_SUCCESS);
135  g_lpspiStatePtr[instance] = lpspiState;
136 
137  /* Configure registers */
138  LPSPI_Init(base);
139 
140  /* Configure lpspi to slave mode */
141  (void)LPSPI_SetMasterSlaveMode(base, LPSPI_SLAVE);
142  /* Set Pin settings */
143  (void)LPSPI_SetPinConfigMode(base, LPSPI_SDI_IN_SDO_OUT, LPSPI_DATA_OUT_RETAINED, true);
144  /* Calculate the FIFO size for the LPSPI */
145  LPSPI_GetFifoSizes(base, &(lpspiState->fifoSize), NULL);
146 
147  /* Set polarity */
148  (void)LPSPI_SetPcsPolarityMode(base, slaveConfig->whichPcs, slaveConfig->pcsPolarity);
149 
150  /* Write the TCR for this transfer */
151  lpspi_tx_cmd_config_t txCmdCfg = {
152  .frameSize = lpspiState->bitsPerFrame,
153  .width = LPSPI_SINGLE_BIT_XFER,
154  .txMask = false,
155  .rxMask = false,
156  .byteSwap = false,
157  .lsbFirst = slaveConfig->lsbFirst,
158  .clkPhase = slaveConfig->clkPhase,
159  .clkPolarity = slaveConfig->clkPolarity,
160  .whichPcs = slaveConfig->whichPcs
161  };
162 
163  /* Write to the TX CMD register */
164  LPSPI_SetTxCommandReg(base, &txCmdCfg);
165  LPSPI_Enable(base);
166  /* Enable the interrupt source */
167  INT_SYS_EnableIRQ(g_lpspiIrqId[instance]);
168 
169  return errorCode;
170 }

204 status_t LPSPI_DRV_SlaveTransferBlocking(uint32_t instance,
205  const uint8_t *sendBuffer,
206  uint8_t *receiveBuffer,
207  uint16_t transferByteCount,
208  uint32_t timeout)
209 {
210  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
211  DEV_ASSERT(g_lpspiStatePtr[instance] != NULL);
212  lpspi_state_t * state = (lpspi_state_t *)g_lpspiStatePtr[instance];
213  status_t error;
214  status_t osifError;
215  state->isBlocking = true;
216  error = LPSPI_DRV_SlaveTransfer(instance, sendBuffer, receiveBuffer, transferByteCount);
217  if(error != STATUS_SUCCESS)
218  {
219  LPSPI_DRV_DisableTEIEInterrupts(instance);
220  return error;
221  }
222 
223  /* As this is a synchronous transfer, wait until the transfer is complete.*/
224  osifError = OSIF_SemaWait(&(state->lpspiSemaphore), timeout);
225 
226  if (osifError == STATUS_TIMEOUT)
227  {
228  /* Set isBlocking variable to false to avoid dummy semaphore post. */
229  state->isBlocking = false;
230  /* Complete transfer. */
231  (void)LPSPI_DRV_SlaveAbortTransfer(instance);
232  return(STATUS_TIMEOUT);
233  }
234 
235  LPSPI_DRV_DisableTEIEInterrupts(instance);
236 
237  return STATUS_SUCCESS;
238 }

243 status_t LPSPI_DRV_SlaveTransfer(uint32_t instance,
244  const uint8_t *sendBuffer,
245  uint8_t *receiveBuffer,
246  uint16_t transferByteCount)
247 {
248  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
249  DEV_ASSERT(g_lpspiStatePtr[instance] != NULL);
250  DEV_ASSERT(!((sendBuffer == NULL) && (receiveBuffer == NULL)));
251  LPSPI_Type * base = g_lpspiBase[instance];
252  lpspi_state_t * state = (lpspi_state_t *)g_lpspiStatePtr[instance];
253  edma_transfer_size_t dmaTransferSize = EDMA_TRANSFER_SIZE_1B;
254  /* The number of transferred bytes should be divisible by frame size */
255  if ((uint16_t)(transferByteCount % state->bytesPerFrame) != (uint16_t)0)
256  {
257  return STATUS_ERROR;
258  }
259  /* Check if LPSPI module isn't busy */
260  if (state->isTransferInProgress == true)
261  {
262  return STATUS_BUSY;
263  }
264  /* Initialize the status of the current transfer */
265  state->status = LPSPI_TRANSFER_OK;
266  /* Clear all interrupts sources */
267  (void)LPSPI_ClearStatusFlag(base, LPSPI_ALL_STATUS);
268  /* Enable fault interrupts sources */
269  LPSPI_SetIntMode(base,LPSPI_TRANSMIT_ERROR , true);
270  LPSPI_SetIntMode(base,LPSPI_RECEIVE_ERROR , true);
271  if (state->transferType == LPSPI_USING_INTERRUPTS)
272  {
273  state->rxBuff = receiveBuffer;
274  state->txBuff = sendBuffer;
275  state->txCount = transferByteCount;
276  state->rxCount = transferByteCount;
277  state->txFrameCnt = 0;
278  state->rxFrameCnt = 0;
279  state->isPcsContinuous = false;
280  /* Configure watermarks */
281  LPSPI_SetRxWatermarks(base, 0U);
282  LPSPI_SetTxWatermarks(base, 2U);
283 
284  /* Clean RX and TX buffers */
285  LPSPI_SetFlushFifoCmd(base, true, true);
286  /* The second flush command is used to avoid the case when one word is still in shifter. */
287  LPSPI_SetFlushFifoCmd(base, true, true);
288  state->isTransferInProgress = true;
289  /* Enable interrupts for RX and TX only if it's necessary */
290  if(state->txBuff != NULL)
291  {
292  LPSPI_SetIntMode(base,LPSPI_TX_DATA_FLAG , true);
293  }
294  else
295  {
296  state->txCount = 0;
297  }
298  if(state->rxBuff != NULL)
299  {
300  LPSPI_SetIntMode(base, LPSPI_RX_DATA_FLAG, true);
301  }
302  else
303  {
304  state->rxCount = 0;
305  }
306  }
307  else
308  {
309  /* Configure watermarks */
310  LPSPI_SetRxWatermarks(base, 0U);
311  LPSPI_SetTxWatermarks(base, 3U);
312  /* When LPSPI use DMA frames with 3 bytes size are not accepted. */
313  switch(state->bytesPerFrame)
314  {
315  case 1: dmaTransferSize = EDMA_TRANSFER_SIZE_1B; break;
316  case 2: dmaTransferSize = EDMA_TRANSFER_SIZE_2B; break;
317  case 4: dmaTransferSize = EDMA_TRANSFER_SIZE_4B; break;
318  default: dmaTransferSize = EDMA_TRANSFER_SIZE_1B; break;
319  }
320  if(receiveBuffer != NULL)
321  {
322  (void)EDMA_DRV_ConfigMultiBlockTransfer(state->rxDMAChannel, EDMA_TRANSFER_PERIPH2MEM,
323  (uint32_t)(&(base->RDR)),(uint32_t)receiveBuffer, dmaTransferSize, (uint32_t)1U<<(uint8_t)(dmaTransferSize),
324  (uint32_t)transferByteCount/(uint32_t)((uint32_t)1U <<(uint8_t)(dmaTransferSize)), true);
325 
326  state->rxCount = transferByteCount;
327  /* Start RX channel */
328  (void)EDMA_DRV_StartChannel(state->rxDMAChannel);
329  }
330  else
331  {
332  state->rxCount = 0;
333  }
334  if(sendBuffer != NULL)
335  {
336  (void)EDMA_DRV_ConfigMultiBlockTransfer(state->txDMAChannel, EDMA_TRANSFER_MEM2PERIPH,
337  (uint32_t)sendBuffer, (uint32_t)(&(base->TDR)), dmaTransferSize, (uint32_t)1U<<(uint8_t)(dmaTransferSize),
338  (uint32_t)transferByteCount/(uint32_t)((uint32_t)1U <<(uint8_t)(dmaTransferSize)), true);
339 
340  state->txCount = transferByteCount;
341  /* Start TX channel */
342  (void)EDMA_DRV_StartChannel(state->txDMAChannel);
343  }
344  else
345  {
346  state->txCount = 0;
347  }
348  /* Configure which channel will generate transfer complete */
349  /* If current transfer uses both buffers (RX and TX) RX transfer done will generate transfer complete
350  * interrupt. Otherwise transfer complete will be generate by available channel(RX or TX).
351  */
352  if(receiveBuffer != NULL)
353  {
354  (void)EDMA_DRV_InstallCallback(state->rxDMAChannel, (LPSPI_DRV_SlaveCompleteDMATransfer),(void*)(instance));
355  }
356  else
357  {
358  (void)EDMA_DRV_InstallCallback(state->txDMAChannel, (LPSPI_DRV_SlaveCompleteDMATransfer),(void*)(instance));
359  }
360  state->isTransferInProgress = true;
361  /* Enable LPSPI DMA request */
362  if (receiveBuffer != NULL)
363  {
364  LPSPI_SetRxDmaCmd(base, true);
365  }
366  if (sendBuffer != NULL)
367  {
368  LPSPI_SetTxDmaCmd(base, true);
369  }
370  }
371  return STATUS_SUCCESS;
372 }

374 void LPSPI_DRV_SlaveIRQHandler(uint32_t instance)
375 {
376  LPSPI_Type * base = g_lpspiBase[instance];
377  lpspi_state_t * lpspiState = (lpspi_state_t *)g_lpspiStatePtr[instance];
378  uint16_t txCount, rxCount;
379 
380  /* If an error is detected the transfer will be aborted */
381  if ((bool)LPSPI_GetStatusFlag(base, LPSPI_TRANSMIT_ERROR) && (lpspiState->txBuff != NULL))
382  {
383  lpspiState->status = LPSPI_TRANSMIT_FAIL;
384  (void)LPSPI_DRV_SlaveAbortTransfer(instance);
385  return;
386  }
387  if (LPSPI_GetStatusFlag(base, LPSPI_RECEIVE_ERROR) && (lpspiState->rxBuff != NULL))
388  {
389  lpspiState->status = LPSPI_RECEIVE_FAIL;
390  (void)LPSPI_DRV_SlaveAbortTransfer(instance);
391  return;
392  }
393 
394  /* Receive data */
395  if(LPSPI_GetStatusFlag(base,LPSPI_RX_DATA_FLAG))
396  {
397  LPSPI_DRV_ReadRXBuffer(instance);
398  }
399  /* Transmit data */
400  txCount = lpspiState->txCount;
401  if (LPSPI_GetStatusFlag(base,LPSPI_TX_DATA_FLAG) && ((txCount != (uint8_t)0)))
402  {
403  LPSPI_DRV_FillupTxBuffer(instance);
404  }
405  txCount = lpspiState->txCount;
406  rxCount = lpspiState->rxCount;
407  /* If all bytes are sent disable interrupt TDF */
408  if (txCount == (uint8_t)0)
409  {
410  LPSPI_SetIntMode(base, LPSPI_TX_DATA_FLAG, false);
411  }
412  /* If all bytes are received disable interrupt RDF */
413  if (rxCount == (uint8_t)0)
414  {
415  LPSPI_SetIntMode(base, LPSPI_RX_DATA_FLAG, false);
416  }
417  if ((rxCount == (uint8_t)0) && (txCount == (uint8_t)0))
418  {
419  /* Disable fault interrupts sources */
420  LPSPI_SetIntMode(base,LPSPI_TRANSMIT_ERROR , false);
421  LPSPI_SetIntMode(base,LPSPI_RECEIVE_ERROR , false);
422 
423  /* Call the callback if it is defined */
424  if (lpspiState->callback != NULL)
425  {
426  lpspiState->callback(lpspiState, SPI_EVENT_END_TRANSFER, lpspiState->callbackParam);
427  }
428 
429  /* If the transfer is blocking post the semaphore */
430  if(lpspiState->isBlocking == true)
431  {
432  (void)OSIF_SemaPost(&(lpspiState->lpspiSemaphore));
433  lpspiState->isBlocking = false;
434  }
435 
436  /* Update internal state of the transfer */
437  lpspiState->isTransferInProgress = false;
438  }
439 }

444 status_t LPSPI_DRV_SlaveAbortTransfer(uint32_t instance)
445 {
446  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
447  DEV_ASSERT(g_lpspiStatePtr[instance] != NULL);
448  LPSPI_Type * base = g_lpspiBase[instance];
449  lpspi_state_t * state = (lpspi_state_t *)g_lpspiStatePtr[instance];
450 
451  if (state->transferType == LPSPI_USING_INTERRUPTS)
452  {
453  /* Disable interrupts */
454  LPSPI_SetIntMode(base, LPSPI_TX_DATA_FLAG, false);
455  LPSPI_SetIntMode(base, LPSPI_RX_DATA_FLAG, false);
456  }
457  else
458  {
459  /* Disable LPSPI DMA request */
460  LPSPI_SetRxDmaCmd(base, false);
461  LPSPI_SetTxDmaCmd(base, false);
462  }
463 
464  LPSPI_DRV_DisableTEIEInterrupts(instance);
465 
466  state->isTransferInProgress = false;
467  /* Clean RX and TX buffers */
468  LPSPI_SetFlushFifoCmd(base, true, true);
469  /* The second flush command is used to avoid the case when one word is still in shifter. */
470  LPSPI_SetFlushFifoCmd(base, true, true);
471  if(state->isBlocking == true)
472  {
473  (void)OSIF_SemaPost(&(state->lpspiSemaphore));
474  state->isBlocking = false;
475  }
476  return STATUS_SUCCESS;
477 }
