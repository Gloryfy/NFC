50 /* For HW triggered groups with a single conversion */
51 void BCTU_ConvUpdate_IRQHandler(void)
52 {
53  uint8_t adcIdx = 0u;
54  const adc_pal_instance_t instance = ADC_PAL_INSTANCE_0;
55 
56  for(adcIdx = 0u; adcIdx < FEATURE_BCTU_NUM_ADC; adcIdx++)
57  {
58  /* ADC PAL doesn't trigger conversions in parallel on multiple ADCs,
59  * so only one ADC should have new data available. */
60  if(BCTU_DRV_GetStatusFlag(instance, adcIdx, BCTU_FLAG_NEW_DATA_AVAILABLE))
61  {
62  ADC_MPC574xC_G_R_HwTrigIrqHandler(instance, adcIdx);
63 
64  break;
65  }
66  }
67 }
