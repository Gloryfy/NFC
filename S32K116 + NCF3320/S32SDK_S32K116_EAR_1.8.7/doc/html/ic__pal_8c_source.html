51 static uint8_t icAllocateState(bool * isAllocated,
52  uint32_t * instanceMapping,
53  uint32_t instance)
54 {
55  uint8_t i;
56 
57  /* Allocate one of the IC state structure for this instance */
58  for (i = 0U;i < NUMBER_OF_IC_PAL_INSTANCES;i++)
59  {
60  if (isAllocated[i] == false)
61  {
62  instanceMapping[i] = instance;
63  isAllocated[i] = true;
64  break;
65  }
66  }
67 
68  return i;
69 }

126  const ic_config_t * configPtr)
127 {
128  DEV_ASSERT(configPtr != NULL);
129  DEV_ASSERT(instance < IC_PAL_INSTANCE_MAX);
130  ic_pal_state_t * icState;
131  status_t status = STATUS_SUCCESS;
132  uint8_t index = 0U;
133  uint8_t indexInstance = 0U;
134 
135  /* Allocate one of the IC state structure for this instance */
136  indexInstance = icAllocateState(icStateIsAllocated, icStateInstanceMapping, instance);
137  icState = &g_icPalStatePtr[indexInstance];
138  DEV_ASSERT(icState->nNumChannels == 0x0U);
139 
140  /* Initialize the internal context to default value */
141  icState->nNumChannels = configPtr->nNumChannels;
142  for (index = 0U; index < IC_PAL_NUM_OF_CHANNEL_MAX; index++)
143  {
144  icState->channelConfigArray[index] = 0U;
145  icState->icChannelType[index] = IC_DISABLE_OPERATION;
146  icState->enableNotification[index] = false;
147  #if (defined(IC_PAL_OVER_FTM))

209  }
210  else if (configPtr->inputChConfig[index].inputCaptureMode == IC_TIMESTAMP_FALLING_EDGE)
211  {
212  channelConfig[index].edgeAlignement = FTM_FALLING_EDGE;
213  }
214  else
215  {
216  channelConfig[index].edgeAlignement = FTM_BOTH_EDGES;
217  }
218  }
219  else
220  {
221  /* Check the channel ID need to even number in the measurement mode */
222  DEV_ASSERT((channel % 2U) == 0U);
223  channelConfig[index].inputMode = FTM_SIGNAL_MEASUREMENT;
224  channelConfig[index].edgeAlignement = FTM_NO_PIN_CONTROL;
225  /* Check the type of measurement */
226  if (configPtr->inputChConfig[index].inputCaptureMode == IC_MEASURE_RISING_EDGE_PERIOD)
227  {
228  channelConfig[index].measurementType = FTM_RISING_EDGE_PERIOD_MEASUREMENT;
229  }

231  {
232  channelConfig[index].measurementType = FTM_FALLING_EDGE_PERIOD_MEASUREMENT;
233  }
234  else if (configPtr->inputChConfig[index].inputCaptureMode == IC_MEASURE_PULSE_HIGH)

240  channelConfig[index].measurementType = FTM_PERIOD_OFF_MEASUREMENT;

246  channelConfig[index].filterValue = configPtr->inputChConfig[index].filterValue;

248  channelConfig[index].channelsCallbacksParams = configPtr->inputChConfig[index].channelCallbackParams;
249  channelConfig[index].channelsCallbacks = configPtr->inputChConfig[index].channelCallbacks;

251  /* Store some needed information into state structure */
252  icState->channelConfigArray[index] = configPtr->inputChConfig[index].hwChannelId;
253  icState->icChannelType[channel] = configPtr->inputChConfig[index].inputCaptureMode;
254  icState->enableContinuousMode[channel] = channelConfig[index].continuousModeEn;
255  icState->enableNotification[channel] = true;
256  }
257 
258  param.inputChConfig = channelConfig;
259  /* Configure channels in input capture mode */
260  status = FTM_DRV_InitInputCapture(instance,
261  &param);
262 
263 #endif

266  emios_common_param_t commonParam;

275  commonParam.lowPowerMode = false;

277  commonParam.enableGlobalPrescaler = ((extension_emios_for_ic_t *)(configPtr->extension))->enableGlobalPrescaler;
278  commonParam.enableGlobalTimeBase = ((extension_emios_for_ic_t *)(configPtr->extension))->enableGlobalTimeBase;
279 
280  /* Initialize the global for a eMIOS group */
281  EMIOS_DRV_InitGlobal((uint8_t)instance,
282  &commonParam);
283 
284  /* Get the information from user configuration */
285  mcParam.mode = EMIOS_MODE_MCB_UP_COUNTER_INT_CLK;
286  mcParam.period = MAX_COUNTER_VALUE;
287  mcParam.filterInput = EMIOS_INPUT_FILTER_BYPASS;
288  mcParam.filterEn = false;
289  mcParam.triggerMode = EMIOS_TRIGGER_EDGE_ANY;
290 
291  /* Configure a list of channels which are used */
292  for (index = 0U; index < configPtr->nNumChannels; index++)
293  {
294  mcParam.internalPrescaler = (emios_clock_internal_ps_t)((channel_extension_emios_for_ic_t *)(configPtr->inputChConfig[index].channelExtension))->prescaler;
295  mcParam.internalPrescalerEn = true;
296 
297  switch (((channel_extension_emios_for_ic_t *)(configPtr->inputChConfig[index].channelExtension))->timebase)
298  {
299  case IC_BUS_SEL_A:
300  /* Set channel to use as a time base */
301  channel = 23U;
302  timeBase = EMIOS_BUS_SEL_A;
303  break;
304  case IC_BUS_SEL_B:
305  /* Set channel to use as a time base */
306  channel = 0U;
307  timeBase = EMIOS_BUS_SEL_BCDE;
308  break;
309  case IC_BUS_SEL_C:
310  /* Set channel to use as a time base */
311  channel = 8U;
312  timeBase = EMIOS_BUS_SEL_BCDE;
313  break;
314  case IC_BUS_SEL_D:
315  /* Set channel to use as a time base */
316  channel = 16U;
317  timeBase = EMIOS_BUS_SEL_BCDE;
318  break;
319  case IC_BUS_SEL_E:

321  channel = 24U;
322  timeBase = EMIOS_BUS_SEL_BCDE;
323  break;
324  case IC_BUS_SEL_F:
325  /* Set channel to use as a time base */
326  channel = 22U;
327  timeBase = EMIOS_BUS_SEL_F;
328  break;

335  /* Do nothing */

339  /* Initialize the counter mode */

341  channel,
342  &mcParam);

344  channel = configPtr->inputChConfig[index].hwChannelId;
345  icParam.mode = EMIOS_MODE_IC;
346  icParam.timebase = timeBase;
347  icParam.filterInput = (emios_input_filter_t)(configPtr->inputChConfig[index].filterValue);
348  icParam.filterEn = configPtr->inputChConfig[index].filterEn;
349 
350  if (configPtr->inputChConfig[index].inputCaptureMode == IC_TIMESTAMP_RISING_EDGE)
351  {
352  icParam.inputCaptureMode = EMIOS_CAPTURE_TRIGGER_EDGE_RISING;
353  }
354  else if (configPtr->inputChConfig[index].inputCaptureMode == IC_TIMESTAMP_FALLING_EDGE)
355  {
356  icParam.inputCaptureMode = EMIOS_CAPTURE_TRIGGER_EDGE_FALLING;
357  }
358  else if (configPtr->inputChConfig[index].inputCaptureMode == IC_DISABLE_OPERATION)
359  {
360  /* Set default configure for an input pin */
361  gpioParam.mode = EMIOS_MODE_GPIO_INPUT;
362  gpioParam.filterEn = false;
363  gpioParam.filterInput = EMIOS_INPUT_FILTER_BYPASS;
364  gpioParam.triggerMode = EMIOS_TRIGGER_EDGE_ANY;
365 
366  /* Disable operation on the channel input */
367  EMIOS_DRV_InitGpioMode((uint8_t)instance,
368  channel,
369  &gpioParam);
370  }
371  else
372  {
373  icParam.inputCaptureMode = (emios_input_capture_mode_t)((uint8_t)configPtr->inputChConfig[index].inputCaptureMode - 1U);
374  }
375 
376  if (configPtr->inputChConfig[index].inputCaptureMode != IC_DISABLE_OPERATION)
377  {
378  /* Initialize the input capture mode for each channel */
379  status = EMIOS_DRV_IC_InitInputCaptureMode((uint8_t)instance,
380  channel,
381  &icParam);
382  }
383 
384  /* Store some needed information into state structure */
385  icState->channelConfigArray[index] = channel;
386  icState->icChannelType[channel] = configPtr->inputChConfig[index].inputCaptureMode;
387  icState->timeBaseSelection[channel] = icParam.timebase;
388  icState->filterEn[channel] = icParam.filterEn;
389  icState->filterInput[channel] = icParam.filterInput;
390  }
391 
392  /* Enable the global eMIOS to start counter */
393  EMIOS_DRV_EnableGlobalEmios((uint8_t)instance);
394 #endif
395 
396  return status;
397 }

399 /*FUNCTION**********************************************************************
400  *
401  * Function Name : IC_Deinit
402  * Description : This function will disable the input capture mode. The driver
403  * can't be used again until reinitialized. The context structure is no longer
404  * needed by the driver and can be freed after calling this function.
405  *
406  * Implements : IC_Deinit_Activity
407  *END**************************************************************************/
408 status_t IC_Deinit(uint32_t instance)
409 {
410  DEV_ASSERT(instance < IC_PAL_INSTANCE_MAX);
411  status_t status = STATUS_SUCCESS;
412  uint8_t index = 0U;
413  ic_pal_state_t * icState;
414 
415  /* Allocate one of the IC state structure for this instance */
416  index = FindIcState(instance);
417  icState = &g_icPalStatePtr[index];
418 
419 #if (defined(IC_PAL_OVER_FTM))
420  /* Disable the input capture over FTM */
421  status = FTM_DRV_Deinit(instance);
422  ftm_state_t * ftmState = &g_ftmState[index];
423 #endif
424 
425 #if (defined(IC_PAL_OVER_EMIOS))
426  uint8_t channel;
427 
428  for (index = 0U; index < icState->nNumChannels; index++)
429  {
430  channel = icState->channelConfigArray[index];
431  /* Disable channels in the input capture over EMIOS */
432  EMIOS_DRV_DeInitChannel((uint8_t)instance,
433  channel);
434  }
435 
436  /* Disable the global of EMIOS */
437  EMIOS_DRV_DisableGlobalEmios((uint8_t)instance);
438 #endif
439 
440  /* De-Initialize the internal context to default value */
441  icState->nNumChannels = 0U;
442  for (index = 0U; index < IC_PAL_NUM_OF_CHANNEL_MAX; index++)
443  {
444  icState->channelConfigArray[index] = 0U;
445  icState->icChannelType[index] = IC_DISABLE_OPERATION;
446  icState->enableNotification[index] = false;
447  #if (defined(IC_PAL_OVER_FTM))
448  icState->enableContinuousMode[index] = false;
449  ftmState->enableNotification[index] = false;
450  #endif
451  }
452 
453  if (status == STATUS_SUCCESS)
454  {
455  /* De-Allocate the instance which is not used */
456  icFreeState(icStateIsAllocated, icStateInstanceMapping, instance);
457  }
458 
459  return status;
460 }

523 /*FUNCTION**********************************************************************
524  *
525  * Function Name : IC_SetChannelMode
526  * Description : This function is used to change the channel mode at run time or
527  * when stopping channel. The channel mode is selected in the ic_option_mode_t
528  * enumeration type.
529  *
530  * Implements : IC_SetChannelMode_Activity
531  *END**************************************************************************/
532 status_t IC_SetChannelMode(uint32_t instance,
533  uint8_t channel,
534  ic_option_mode_t channelMode)
535 {
536  DEV_ASSERT(instance < IC_PAL_INSTANCE_MAX);
537  DEV_ASSERT(channel < IC_PAL_NUM_OF_CHANNEL_MAX);
538  uint8_t index = 0U;
539  ic_pal_state_t * icState;
540  status_t status = STATUS_SUCCESS;
541 
542  /* Allocate one of the IC state structure for this instance */
543  index = FindIcState(instance);
544  icState = &g_icPalStatePtr[index];
545 
546 #if (defined(IC_PAL_OVER_FTM))
547  bool contModeEnable;
548 
549  if (true == icState->enableContinuousMode[channel])
550  {
551  contModeEnable = true;
552  }
553  else
554  {
555  contModeEnable = false;
556  }
557 
558  /* Set operation mode for channel input */
559  status = FTM_IC_DRV_SetChannelMode(instance,
560  channel,
561  (ftm_ic_op_mode_t)channelMode,
562  contModeEnable);
563 #endif
564 
565 #if (defined(IC_PAL_OVER_EMIOS))
566  emios_input_capture_param_t icParam;
567  emios_gpio_mode_param_t gpioParam;
568 
569  if (channelMode == IC_DISABLE_OPERATION)
570  {
571  /* Set default configure for an input pin */
572  gpioParam.mode = EMIOS_MODE_GPIO_INPUT;
573  gpioParam.filterEn = false;
574  gpioParam.filterInput = EMIOS_INPUT_FILTER_BYPASS;
575  gpioParam.triggerMode = EMIOS_TRIGGER_EDGE_ANY;
576 
577  /* Disable operation on the channel input */
578  EMIOS_DRV_InitGpioMode((uint8_t)instance,
579  channel,
580  &gpioParam);
581  }
582  else
583  {
584  icParam.mode = EMIOS_MODE_IC;
585  icParam.timebase = icState->timeBaseSelection[channel];
586  icParam.filterInput = icState->filterInput[channel];
587  icParam.filterEn = icState->filterEn[channel];
588 
589  if (channelMode == IC_TIMESTAMP_RISING_EDGE)
590  {
591  icParam.inputCaptureMode = EMIOS_CAPTURE_TRIGGER_EDGE_RISING;
592  }
593  else if (channelMode == IC_TIMESTAMP_FALLING_EDGE)
594  {
595  icParam.inputCaptureMode = EMIOS_CAPTURE_TRIGGER_EDGE_FALLING;
596  }
597  else
598  {
599  icParam.inputCaptureMode = (emios_input_capture_mode_t)((uint8_t)(channelMode) - 1U);
600  }
601 
602  /* Initialize the input capture mode for each channel */
603  status = EMIOS_DRV_IC_InitInputCaptureMode((uint8_t)instance,
604  channel,
605  &icParam);
606  }
607 #endif
608 
609  /* Update the channel mode */
610  icState->icChannelType[channel] = channelMode;
611 
612  return status;
613 }
