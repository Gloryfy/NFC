125 /*FUNCTION**********************************************************************
126  *
127  * Function Name : LPSPI_DRV_MasterInit
128  * Description : Initializes a LPSPI instance for interrupt driven master mode operation.
129  *
130  * This function uses an interrupt-driven method for transferring data.
131  * In this function, the term "spiConfig" is used to indicate the SPI device for which the LPSPI
132  * master is communicating.
133  * This function initializes the run-time state structure to track the ongoing
134  * transfers, un-gates the clock to the LPSPI module, resets the LPSPI module,
135  * configures the IRQ state structure, enables the module-level interrupt to the core, and
136  * enables the LPSPI module.
137  * Implements : LPSPI_DRV_MasterInit_Activity
138  *
139  *END**************************************************************************/
140 status_t LPSPI_DRV_MasterInit(uint32_t instance, lpspi_state_t * lpspiState,
141  const lpspi_master_config_t * spiConfig)
142 {
143  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
144  DEV_ASSERT(lpspiState != NULL);
145  DEV_ASSERT(spiConfig != NULL);
146  LPSPI_Type *base = g_lpspiBase[instance];
147  status_t errorCode = STATUS_SUCCESS;
148 
149  /* Save runtime structure pointers so irq handler can point to the correct state structure */
150  g_lpspiStatePtr[instance] = lpspiState;
151  /* Reset the LPSPI registers to their default state */
152  LPSPI_Init(base);
153  /* Set for master mode */
154  (void)LPSPI_SetMasterSlaveMode(base, LPSPI_MASTER);
155  /* Set Pin configuration such that SDO=out and SDI=in */
156  (void)LPSPI_SetPinConfigMode(base, LPSPI_SDI_IN_SDO_OUT, LPSPI_DATA_OUT_RETAINED, true);
157  /* Calculate the FIFO size for the LPSPI */
158  LPSPI_GetFifoSizes(base, &(lpspiState->fifoSize), NULL);
159 
160  /* Configure bus for this device. If NULL is passed, we assume the caller has
161  * preconfigured the bus and doesn't wish to re-configure it again for this transfer.
162  * Do nothing for calculatedBaudRate. If the user wants to know the calculatedBaudRate
163  * then they can call this function separately.
164  */
165  errorCode = LPSPI_DRV_MasterConfigureBus(instance, spiConfig, NULL);
166  if (errorCode != STATUS_SUCCESS)
167  {
168  return errorCode;
169  }
170  /* When TX is null the value sent on the bus will be 0 */
171  lpspiState->dummy = 0;
172  /* Initialize the semaphore */
173  errorCode = OSIF_SemaCreate(&(lpspiState->lpspiSemaphore), 0);
174  DEV_ASSERT(errorCode == STATUS_SUCCESS);
175  /* Enable the interrupt */
176  INT_SYS_EnableIRQ(g_lpspiIrqId[instance]);
177  /* Finally, enable LPSPI */
178  LPSPI_Enable(base);
179  return errorCode;
180 }

182 /*FUNCTION**********************************************************************
183  *
184  * Function Name : LPSPI_DRV_MasterDeinit
185  * Description : Shuts down a LPSPI instance.
186  *
187  * This function resets the LPSPI peripheral, gates its clock, and disables the interrupt to
188  * the core. It first checks to see if a transfer is in progress and if so returns an error
189  * status.
190  * Implements : LPSPI_DRV_MasterDeinit_Activity
191  *
192  *END**************************************************************************/
193 status_t LPSPI_DRV_MasterDeinit(uint32_t instance)
194 {
195  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
196  DEV_ASSERT(g_lpspiStatePtr[instance] != NULL);
197  /* Instantiate local variable of type lpspi_state_t and point to global state */
198  const lpspi_state_t * lpspiState = g_lpspiStatePtr[instance];
199  LPSPI_Type *base = g_lpspiBase[instance];
200  status_t errorCode = STATUS_SUCCESS;
201 
202  /* Check if a transfer is still in progress */
203  DEV_ASSERT(lpspiState->isTransferInProgress == false);
204 
205  /* Reset the LPSPI registers to their default state, inlcuding disabling the LPSPI */
206  LPSPI_Init(base);
207  /* Disable the interrupt */
208  INT_SYS_DisableIRQ(g_lpspiIrqId[instance]);
209  /* Clear the state pointer. */
210  g_lpspiStatePtr[instance] = NULL;
211 
212  /* Destroy the semaphore */
213  errorCode = OSIF_SemaDestroy(&(lpspiState->lpspiSemaphore));
214  DEV_ASSERT(errorCode == STATUS_SUCCESS);
215  return errorCode;

218 /*FUNCTION**********************************************************************
219  *
220  * Function Name : LPSPI_DRV_MasterSetDelay
221  * Description : Configures the LPSPI master mode bus timing delay options.
222  *
223  * This function involves the LPSPI module's delay options to
224  * "fine tune" some of the signal timings and match the timing needs of a slower peripheral device.
225  * This is an optional function that can be called after the LPSPI module has been initialized for
226  * master mode. The timings are adjusted in terms of cycles of the baud rate clock.
227  * The bus timing delays that can be adjusted are listed below:
228  *
229  * SCK to PCS Delay: Adjustable delay option between the last edge of SCK to the de-assertion
230  * of the PCS signal.
231  *
232  * PCS to SCK Delay: Adjustable delay option between the assertion of the PCS signal to the
233  * first SCK edge.
234  *
235  * Delay between Transfers: Adjustable delay option between the de-assertion of the PCS signal for
236  * a frame to the assertion of the PCS signal for the next frame.
237  * Implements : LPSPI_DRV_MasterSetDelay_Activity
238  *
239  *END**************************************************************************/
240 status_t LPSPI_DRV_MasterSetDelay(uint32_t instance, uint32_t delayBetwenTransfers,
241  uint32_t delaySCKtoPCS, uint32_t delayPCStoSCK)
242 {
243  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
244  DEV_ASSERT(g_lpspiStatePtr[instance] != NULL);
245  /* Instantiate local variable of type lpspi_state_t and point to global state */
246  const lpspi_state_t * lpspiState = g_lpspiStatePtr[instance];
247  LPSPI_Type *base = g_lpspiBase[instance];
248  status_t errorCode = STATUS_SUCCESS;
249  uint32_t realDelayBetwenTransfers, realDelaySCKtoPCS, realDelayPCStoSCK;
250  lpspi_prescaler_t prescaler;
251  /* Disable module */
252  errorCode = LPSPI_Disable(base);
253  if (errorCode != STATUS_SUCCESS)
254  {
255  return errorCode;
256  }
257  prescaler = LPSPI_GetClockPrescaler(base);
258  realDelayBetwenTransfers = delayBetwenTransfers * lpspiState->lpspiSrcClk / s_baudratePrescaler[prescaler] / (uint32_t)MICROSECONDS;
259  realDelaySCKtoPCS = delaySCKtoPCS * lpspiState->lpspiSrcClk / s_baudratePrescaler[prescaler] / (uint32_t)MICROSECONDS;
260  realDelayPCStoSCK = delayPCStoSCK * lpspiState->lpspiSrcClk/ s_baudratePrescaler[prescaler] / (uint32_t)MICROSECONDS;
261  /* Verify if current prescaler can be used for this configuration of delays
262  * If a delay is out of range, it will be adjusted to min or max value */
263  if (realDelayBetwenTransfers > (uint32_t)257U)
264  {
265  realDelayBetwenTransfers = (uint32_t)257U;
266  }
267  if(realDelaySCKtoPCS > (uint32_t)256U)
268  {
269  realDelaySCKtoPCS = (uint32_t)256U;
270  }
271  if(realDelayPCStoSCK > (uint32_t)256U)
272  {
273  realDelayPCStoSCK = (uint32_t)256U;
274  }
275  if (realDelayBetwenTransfers < (uint32_t)2U)
276  {
277  realDelayBetwenTransfers = (uint32_t)2U;
278  }
279  if(realDelaySCKtoPCS == (uint32_t)0)
280  {
281  realDelaySCKtoPCS = (uint32_t)1U;
282  }
283  if(realDelayPCStoSCK == (uint32_t)0U)
284  {
285  realDelayPCStoSCK = (uint32_t)1U;
286  }
287 
288  (void)LPSPI_SetDelay(base, LPSPI_SCK_TO_PCS, realDelaySCKtoPCS-1U);
289  (void)LPSPI_SetDelay(base, LPSPI_PCS_TO_SCK, realDelayPCStoSCK-1U);
290  (void)LPSPI_SetDelay(base, LPSPI_BETWEEN_TRANSFER, realDelayBetwenTransfers-2U);
291  /* Enable module */
292  LPSPI_Enable(base);
293  return errorCode;
294 }

296 /*FUNCTION**********************************************************************
297  *
298  * Function Name : LPSPI_DRV_MasterConfigureBus
299  * Description : Configures the LPSPI port physical parameters to access a device on the bus when
300  * the LSPI instance is configured for interrupt operation.
301  *
302  * In this function, the term "spiConfig" is used to indicate the SPI device for which the LPSPI
303  * master is communicating. This is an optional function as the spiConfig parameters are
304  * normally configured in the initialization function or the transfer functions, where these various
305  * functions would call the configure bus function.
306  * The user can pass in a different spiConfig structure to the transfer function which contains
307  * the parameters for the SPI bus to allow for communication to a different SPI device
308  * (the transfer function then calls this function). However, the user also has the option to call
309  * this function directly especially to get the calculated baud rate, at which point they may pass
310  * in NULL for the spiConfig structure in the transfer function (assuming they have called this
311  * configure bus function first).
312  * Implements : LPSPI_DRV_MasterConfigureBus_Activity
313  *
314  *END**************************************************************************/
315 status_t LPSPI_DRV_MasterConfigureBus(uint32_t instance,
316  const lpspi_master_config_t * spiConfig,
317  uint32_t * calculatedBaudRate)
318 {
319  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
320  DEV_ASSERT(spiConfig != NULL);
321  DEV_ASSERT(g_lpspiStatePtr[instance] != NULL);
322  /* Instantiate local variable of type lpspi_state_t and point to global state */
323  lpspi_state_t * lpspiState = g_lpspiStatePtr[instance];
324  LPSPI_Type *base = g_lpspiBase[instance];
325  uint32_t baudRate;
326 
327  /* The Transmit Command Register (TCR) Prescale value is calculated as part of the baud rate
328  calculation. The value is stored in the run-time state structure for later programming
329  in the TCR. */
330  uint32_t tcrPrescaleValue;
331 
332  /* Check the bitcount to make sure it falls within the boundary conditions */
333  if ((spiConfig->bitcount < 8U) || (spiConfig->bitcount > 4096U))
334  {
335  return STATUS_ERROR;
336  }
337  /* First, per the spec, we need to disable the LPSPI module before setting the delay */
338 
339  if (LPSPI_Disable(base) != STATUS_SUCCESS)
340  {
341  /* If error is returned, the LPSPI is busy */
342  return STATUS_ERROR;
343  }
344  /* Configure internal state structure for LPSPI */
345  lpspiState->bitsPerFrame = spiConfig->bitcount;
346  lpspiState->lpspiSrcClk = spiConfig->lpspiSrcClk;
347  lpspiState->isPcsContinuous = spiConfig->isPcsContinuous;
348  lpspiState->lsb = spiConfig->lsbFirst;
349  /* Save transfer type DMA/Interrupt */
350  lpspiState->transferType = spiConfig->transferType;
351  /* Update transfer status */
352  lpspiState->isTransferInProgress = false;
353  lpspiState->isBlocking = false;
354  /* Calculate the bytes/frame for lpspiState->bytesPerFrame. */
355  lpspiState->bytesPerFrame = (uint16_t)((lpspiState->bitsPerFrame + 7U) / 8U);
356  /* For DMA transfers bytes per frame must be equal to 1, 2 or multiple of 4 */
357  if ((lpspiState->transferType == LPSPI_USING_DMA) && (!(((lpspiState->bytesPerFrame % 4U) == (uint16_t)0) ||
358  (lpspiState->bytesPerFrame<=2U))))
359  {
360  return STATUS_ERROR;
361  }
362  /* Store DMA channel number used in transfer */
363  lpspiState->rxDMAChannel = spiConfig->rxDMAChannel;
364  lpspiState->txDMAChannel = spiConfig->txDMAChannel;
365  /* Store callback */
366  lpspiState->callback = spiConfig->callback;
367  lpspiState->callbackParam = spiConfig->callbackParam;
368  /* Configure the desired PCS polarity */
369  (void)LPSPI_SetPcsPolarityMode(base, spiConfig->whichPcs, spiConfig->pcsPolarity);
370  /* Set up the baud rate */
371  baudRate = LPSPI_SetBaudRate(base, spiConfig->bitsPerSec, spiConfig->lpspiSrcClk,
372  &tcrPrescaleValue);
373  /* Enable sampling point delay */
374  LPSPI_SetSamplingPoint(base, true);
375  /* Now, re-enable the LPSPI module */
376  LPSPI_Enable(base);
377  /* If the baud rate return is "0", it means there was an error */
378  if (baudRate == (uint32_t)0)
379  {
380  return STATUS_ERROR;
381  }
382  /* If the user wishes to know the calculated baud rate, then pass it back */
383  if (calculatedBaudRate != NULL)
384  {
385  *calculatedBaudRate = baudRate;
386  }
387  /* Write the TCR for this transfer. */
388  lpspi_tx_cmd_config_t txCmdCfg =
389  {
390  .frameSize = lpspiState->bitsPerFrame,
391  .width = LPSPI_SINGLE_BIT_XFER,
392  .txMask = false,
393  .rxMask = false,
394  .contCmd = false,
395  .contTransfer = spiConfig->isPcsContinuous,
396  .byteSwap = false,
397  .lsbFirst = spiConfig->lsbFirst,
398  .whichPcs = spiConfig->whichPcs,
399  .preDiv = tcrPrescaleValue,
400  .clkPhase = spiConfig->clkPhase,
401  .clkPolarity = spiConfig->clkPolarity
402  };
403  LPSPI_SetTxCommandReg(base, &txCmdCfg);
404  return STATUS_SUCCESS;
405 }

407 /*FUNCTION**********************************************************************
408  *
409  * Function Name : LPSPI_DRV_MasterTransferBlocking
410  * Description : Performs an interrupt driven blocking SPI master mode transfer.
411  *
412  * This function simultaneously sends and receives data on the SPI bus, as SPI is naturally
413  * a full-duplex bus. The function does not return until the transfer is complete.
414  * This function allows the user to optionally pass in a SPI configuration structure which
415  * allows the user to change the SPI bus attributes in conjunction with initiating a SPI transfer.
416  * The difference between passing in the SPI configuration structure here as opposed to the
417  * configure bus function is that the configure bus function returns the calculated baud rate where
418  * this function does not. The user can also call the configure bus function prior to the transfer
419  * in which case the user would simply pass in a NULL to the transfer function's device structure
420  * parameter.
421  * Implements : LPSPI_DRV_MasterTransferBlocking_Activity
422  *
423  *END**************************************************************************/
424 status_t LPSPI_DRV_MasterTransferBlocking(uint32_t instance,
425  const uint8_t * sendBuffer,
426  uint8_t * receiveBuffer,
427  uint16_t transferByteCount,
428  uint32_t timeout)
429 {
430  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
431  DEV_ASSERT(g_lpspiStatePtr[instance] != NULL);
432  /* Instantiate local variable of type lpspi_state_t and point to global state */
433  lpspi_state_t * lpspiState = g_lpspiStatePtr[instance];
434  LPSPI_Type *base = g_lpspiBase[instance];
435  status_t error = STATUS_SUCCESS;
436  status_t osifError;
437  /* If the transfer count is zero, then return immediately.*/
438  if (transferByteCount == (uint16_t)0)
439  {
440  return error;
441  }
442  lpspiState->isBlocking = true;
443  /* start the transfer process, if it returns an error code, return this back to user */
444  error = LPSPI_DRV_MasterStartTransfer(instance, sendBuffer, receiveBuffer,
445  transferByteCount);
446  if (error != STATUS_SUCCESS)
447  {
448  /* The transfer is complete.*/
449  lpspiState->isTransferInProgress = false;
450 
451  /* Disable interrupt requests*/
452  LPSPI_SetIntMode(base, LPSPI_TX_DATA_FLAG, false);
453  LPSPI_SetIntMode(base, LPSPI_RX_DATA_FLAG, false);
454 
455  LPSPI_DRV_DisableTEIEInterrupts(instance);
456  LPSPI_SetIntMode(base, LPSPI_TRANSFER_COMPLETE, false);
457  (void)LPSPI_ClearStatusFlag(base, LPSPI_TRANSFER_COMPLETE);
458 
459  return error;
460  }
461 
462  /* As this is a synchronous transfer, wait until the transfer is complete.*/
463  osifError = OSIF_SemaWait(&(lpspiState->lpspiSemaphore), timeout);
464 
465  /* If a timeout occurs, stop the transfer by setting the isTransferInProgress to false and
466  * disabling interrupts, then return the timeout error status.
467  */
468  if (osifError == STATUS_TIMEOUT)
469  {
470  /* Set isBlocking variable to false to avoid dummy semaphore post. */
471  lpspiState->isBlocking = false;
472  /* Complete transfer. */
473  LPSPI_DRV_MasterCompleteTransfer(instance);
474  return(STATUS_TIMEOUT);
475  }
476 
477  LPSPI_DRV_DisableTEIEInterrupts(instance);
478  LPSPI_SetIntMode(base, LPSPI_TRANSFER_COMPLETE, false);
479  (void)LPSPI_ClearStatusFlag(base, LPSPI_TRANSFER_COMPLETE);
480 
481  return error;
482 }

484 /*FUNCTION**********************************************************************
485  *
486  * Function Name : LPSPI_DRV_MasterTransfer
487  * Description : Performs an interrupt driven non-blocking SPI master mode transfer.
488  *
489  * This function simultaneously sends and receives data on the SPI bus, as SPI is naturally
490  * a full-duplex bus. The function returns immediately after initiating the transfer. The user
491  * needs to check whether the transfer is complete using the LPSPI_DRV_MasterGetTransferStatus
492  * function.
493  * This function allows the user to optionally pass in a SPI configuration structure which
494  * allows the user to change the SPI bus attributes in conjunction with initiating a SPI transfer.
495  * The difference between passing in the SPI configuration structure here as opposed to the
496  * configure bus function is that the configure bus function returns the calculated baud rate where
497  * this function does not. The user can also call the configure bus function prior to the transfer
498  * in which case the user would simply pass in a NULL to the transfer function's device structure
499  * parameter.
500  * Implements : LPSPI_DRV_MasterTransfer_Activity
501  *
502  *END**************************************************************************/
503 status_t LPSPI_DRV_MasterTransfer(uint32_t instance,
504  const uint8_t * sendBuffer,
505  uint8_t * receiveBuffer,
506  uint16_t transferByteCount)
507 {
508  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
509  DEV_ASSERT(g_lpspiStatePtr[instance] != NULL);
510  status_t error = STATUS_SUCCESS;
511  /* If the transfer count is zero, then return immediately.*/
512  if (transferByteCount == (uint16_t)0)
513  {
514  return STATUS_SUCCESS;
515  }
516 
517  /* Start the transfer process, if it returns an error code, return this back to user */
518  error = LPSPI_DRV_MasterStartTransfer(instance, sendBuffer, receiveBuffer,
519  transferByteCount);
520  if (error != STATUS_SUCCESS)
521  {
522  return error;
523  }
524 
525  /* Else, return immediately as this is an async transfer */
526  return STATUS_SUCCESS;
527 }

532  * Description : Returns whether the previous interrupt driven transfer is completed.

535  * the state of the current transfer: in progress (or busy) or complete (success).

545  /* Instantiate local variable of type lpspi_state_t and point to global state */
546  const lpspi_state_t * lpspiState = g_lpspiStatePtr[instance];
547  /* Fill in the bytes transferred.*/
548  if (bytesRemained != NULL)
549  {
550  *bytesRemained = lpspiState->rxCount;
551  }
552  if (lpspiState->status == LPSPI_TRANSFER_OK)
553  {
554  return (status_t)(lpspiState->isTransferInProgress ? STATUS_BUSY : STATUS_SUCCESS);
555  }
556  else
557  {
558  return STATUS_ERROR;
559  }
560 }

614 /*FUNCTION**********************************************************************
615  *
616  * Function Name : LPSPI_DRV_MasterStartTransfer
617  * Description : Configure a non-blocking transfer.
618  *
619  * The number of transferByteCount must be divided by number of bytes/frame.
620  * The sendBuffer must be not NULL, but receiveBuffer can be NULL.
621  *

623 static status_t LPSPI_DRV_MasterStartTransfer(uint32_t instance,
624  const uint8_t * sendBuffer,
625  uint8_t * receiveBuffer,
626  uint16_t transferByteCount)
627 {
628  DEV_ASSERT(instance < LPSPI_INSTANCE_COUNT);
629  DEV_ASSERT(g_lpspiStatePtr[instance] != NULL);
630  /* Instantiate local variable of type dspi_master_state_t and point to global state */
631  lpspi_state_t * lpspiState = g_lpspiStatePtr[instance];
632  LPSPI_Type *base = g_lpspiBase[instance];
633  edma_transfer_size_t dmaTransferSize = EDMA_TRANSFER_SIZE_1B;
634 
635  /* Check that we're not busy. */
636  if (LPSPI_GetStatusFlag(base, LPSPI_MODULE_BUSY))
637  {
638  return STATUS_BUSY;
639  }
640  #ifdef ERRATA_E10655
641  else
642  {
643  /* Double check to fix errata e10655. */
644  if (LPSPI_GetStatusFlag(base, LPSPI_MODULE_BUSY))
645  {
646  return STATUS_BUSY;
647  }
648  }
649  #endif
650 
651  /* Verify if the number of bytes is divided by number of bytes/frame. */
652  if ((transferByteCount % lpspiState->bytesPerFrame) != (uint16_t)0)
653  {
654  return STATUS_ERROR;
655  }
656  if(lpspiState->isPcsContinuous == true)
657  {
658  LPSPI_SetContCBit(base);
659  }
660 
661  /* Configure watermarks */
662  LPSPI_SetRxWatermarks(base, 0U);
663  LPSPI_SetTxWatermarks(base, 2U);
664 
665  lpspiState->status = LPSPI_TRANSFER_OK;
666  /* Clear all interrupts sources */
667  (void)LPSPI_ClearStatusFlag(base, LPSPI_ALL_STATUS);
668  /* Enable fault interrupts sources */
669  LPSPI_SetIntMode(base,LPSPI_TRANSMIT_ERROR , true);
670  if (receiveBuffer != NULL)
671  {
672  LPSPI_SetIntMode(base,LPSPI_RECEIVE_ERROR , true);
673  }
674  if (lpspiState->transferType == LPSPI_USING_INTERRUPTS)
675  {
676  /* Fill out the other members of the run-time state structure. */
677  lpspiState->txBuff = (const uint8_t *)sendBuffer;
678  lpspiState->rxBuff = (uint8_t *)receiveBuffer;
679  lpspiState->txFrameCnt = 0;
680  lpspiState->rxFrameCnt = 0;
681  lpspiState->txCount = transferByteCount;
682  /*For continuous mode an extra word must be written to negate the PCS */
683  if (lpspiState->isPcsContinuous == true)
684  {
685  lpspiState->txCount++;
686  }
687  /* Clean RX and TX buffers */
688  LPSPI_SetFlushFifoCmd(base, true, true);
689  /* The second flush command is used to avoid the case when one word is still in shifter. */
690  LPSPI_SetFlushFifoCmd(base, true, true);
691  /* Update transfer status */
692  lpspiState->isTransferInProgress = true;
693  /* Mask the RX if no buffer is passed in. */
694  if (lpspiState->rxBuff == NULL)
695  {
696  /* Since we're not receiving, set rxCount to 0. */
697  lpspiState->rxCount = 0;
698  }
699  else
700  {
701  /* Enable RDF interrupt if RX buffer is not NULL. */
702  LPSPI_SetIntMode(base, LPSPI_RX_DATA_FLAG, true);
703  lpspiState->rxCount = transferByteCount;
704  }
705  /* Enable the TDF and RDF interrupt. */
706  LPSPI_SetIntMode(base, LPSPI_TX_DATA_FLAG, true);
707  }
708  else
709  {
710  /* Configure rxCount depending on transfer type.*/
711  if (receiveBuffer == NULL)
712  {
713  lpspiState->rxCount = 0;
714  }
715  else
716  {
717  lpspiState->rxCount = transferByteCount;
718  }
719 
720  /* When LPSPI use DMA frames with 3 bytes size are not accepted. */
721  switch(lpspiState->bytesPerFrame)
722  {
723  case 1: dmaTransferSize = EDMA_TRANSFER_SIZE_1B; break;
724  case 2: dmaTransferSize = EDMA_TRANSFER_SIZE_2B; break;
725  case 4: dmaTransferSize = EDMA_TRANSFER_SIZE_4B; break;
726  default : dmaTransferSize = EDMA_TRANSFER_SIZE_1B; break;
727  }
728  /* Configure TX DMA channel */
729  if (sendBuffer != NULL)
730  {
731  (void)EDMA_DRV_ConfigMultiBlockTransfer(lpspiState->txDMAChannel, EDMA_TRANSFER_MEM2PERIPH,
732  (uint32_t)sendBuffer, (uint32_t)(&(base->TDR)), dmaTransferSize, (uint32_t)1U<<(uint8_t)(dmaTransferSize),
733  (uint32_t)transferByteCount/(uint32_t)((uint32_t)1U <<(uint8_t)(dmaTransferSize)), true);
734  }
735  else
736  {
737  (void)EDMA_DRV_ConfigMultiBlockTransfer(lpspiState->txDMAChannel, EDMA_TRANSFER_PERIPH2PERIPH,
738  (uint32_t)(&(lpspiState->dummy)), (uint32_t)(&(base->TDR)), dmaTransferSize, (uint32_t)1U<<(uint8_t)(dmaTransferSize),
739  (uint32_t)transferByteCount/(uint32_t)((uint32_t)1U <<(uint8_t)(dmaTransferSize)), true);
740  }
741  /* Start TX channel */
742  (void)EDMA_DRV_StartChannel(lpspiState->txDMAChannel);
743  /* Configure RX DMA channel if is used in current transfer. */
744  if(receiveBuffer != NULL)
745  {
746  (void)EDMA_DRV_ConfigMultiBlockTransfer(lpspiState->rxDMAChannel, EDMA_TRANSFER_PERIPH2MEM,
747  (uint32_t)(&(base->RDR)),(uint32_t)receiveBuffer, dmaTransferSize, (uint32_t)1U<<(uint8_t)(dmaTransferSize),
748  (uint32_t)transferByteCount/(uint32_t)((uint32_t)1U <<(uint8_t)(dmaTransferSize)), true);
749 
750  (void)EDMA_DRV_InstallCallback(lpspiState->rxDMAChannel, (LPSPI_DRV_MasterCompleteDMATransfer),(void*)(instance));
751  if (lpspiState->isPcsContinuous == true)
752  {
753  (void)EDMA_DRV_InstallCallback(lpspiState->txDMAChannel, (LPSPI_DRV_MasterClearCountinuous),(void*)(instance));
754  }
755  /* Start RX channel */
756  (void)EDMA_DRV_StartChannel(lpspiState->rxDMAChannel);
757  }
758  else
759  {
760  /* If RX buffer is null the transfer is done when all bytes were sent. */
761  (void)EDMA_DRV_InstallCallback(lpspiState->txDMAChannel, (LPSPI_DRV_MasterCompleteDMATransfer),(void*)(instance));
762  }
763  /* Update transfer status */
764  lpspiState->isTransferInProgress = true;
765  /* Enable LPSPI DMA request */
766  if (receiveBuffer!=NULL)
767  {
768  LPSPI_SetRxDmaCmd(base, true);
769  }
770  LPSPI_SetTxDmaCmd(base, true);
771  }
772  return STATUS_SUCCESS;
773 }

782  /* instantiate local variable of type dspi_master_state_t and point to global state */
783  lpspi_state_t * lpspiState = g_lpspiStatePtr[instance];

847 void LPSPI_DRV_MasterIRQHandler(uint32_t instance)
848 {
849  /* Instantiate local variable of type dspi_master_state_t and point to global state */
850  lpspi_state_t * lpspiState = g_lpspiStatePtr[instance];
851  LPSPI_Type *base = g_lpspiBase[instance];
852  uint16_t txCount, rxCount;
853  txCount = lpspiState->txCount;
854  rxCount = lpspiState->rxCount;
855 
856  /* If an error is detected the transfer will be aborted */
857  if ((bool)LPSPI_GetStatusFlag(base, LPSPI_TRANSMIT_ERROR) && (lpspiState->txBuff != NULL))
858  {
859  lpspiState->status = LPSPI_TRANSMIT_FAIL;
860  (void)LPSPI_DRV_MasterAbortTransfer(instance);
861  (void)LPSPI_ClearStatusFlag(base, LPSPI_TRANSMIT_ERROR);
862  return;
863  }
864  if (LPSPI_GetStatusFlag(base, LPSPI_RECEIVE_ERROR) && (lpspiState->rxBuff != NULL))
865  {
866  lpspiState->status = LPSPI_RECEIVE_FAIL;
867  (void)LPSPI_DRV_MasterAbortTransfer(instance);
868  (void)LPSPI_ClearStatusFlag(base, LPSPI_RECEIVE_ERROR);
869  return;
870  }
871 
872  /* RECEIVE IRQ handler: Check read buffer only if there are remaining bytes to read. */
873  if(LPSPI_GetStatusFlag(base,LPSPI_RX_DATA_FLAG) && (rxCount != (uint16_t)0))
874  {
875  LPSPI_DRV_ReadRXBuffer(instance);
876  }
877  /* Transmit data */
878  if (LPSPI_GetStatusFlag(base,LPSPI_TX_DATA_FLAG) && ((txCount != (uint16_t)0)))
879  {
880  LPSPI_DRV_FillupTxBuffer(instance);
881  }
882  txCount = lpspiState->txCount;
883  rxCount = lpspiState->rxCount;
884  if (txCount == (uint16_t)0)
885  {
886  /* Disable TX flag. Software buffer is empty.*/
887  LPSPI_SetIntMode(base, LPSPI_TX_DATA_FLAG, false);
888  LPSPI_SetIntMode(base, LPSPI_TRANSFER_COMPLETE, true);
889  /* Check if we're done with this transfer.*/
890  if (rxCount == (uint16_t)0)
891  {
892  if (LPSPI_GetStatusFlag(base, LPSPI_TRANSFER_COMPLETE) == true)
893  {
894  LPSPI_DRV_MasterCompleteTransfer(instance);
895  }
896  }
897  }
