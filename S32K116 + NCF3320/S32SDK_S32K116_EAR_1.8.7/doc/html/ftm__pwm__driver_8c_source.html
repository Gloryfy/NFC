35 /*FUNCTION**********************************************************************
36  *
37  * Function Name : FTM_DRV_InitPwm
38  * Description : Configures duty cycle and frequency and starts outputting
39  * PWM on specified channels.
40  *
41  * Implements : FTM_DRV_InitPwm_Activity
42  *END**************************************************************************/
43 status_t FTM_DRV_InitPwm(uint32_t instance,
44  const ftm_pwm_param_t * param)
45 {
46  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
47  DEV_ASSERT(param != NULL);
48  status_t retVal = STATUS_SUCCESS;
49  uint8_t index = 0U;
50  uint8_t hwChannel = 0U;
51  uint8_t fltChannel = 0U;
52  uint8_t chnlPairNum = 0U;
53  uint8_t channelId = 0U;
54  ftm_state_t * state = ftmStatePtr[instance];
55  FTM_Type * ftmBase = g_ftmBase[instance];
56 
57  if ((NULL != state) && (FTM_MODE_NOT_INITIALIZED == state->ftmMode))
58  {
59  /* Disable counter clock */
60  FTM_DRV_SetClockSource(ftmBase, FTM_CLOCK_SOURCE_NONE);
61  /* Clear the overflow flag */
62  FTM_DRV_ClearTimerOverflow(ftmBase);
63  /* Disable write protection */
64  FTM_DRV_SetWriteProtectionCmd(ftmBase, false);
65  /* Configure FTM mode */
66  state->ftmMode = param->mode;
67 
68  /* Configure independent PWM channels */
69  for (index = 0U; index < param->nNumIndependentPwmChannels; index++)
70  {
71  channelId = param->pwmIndependentChannelConfig[index].hwChannelId;
72  chnlPairNum = (uint8_t)(channelId >> 1U);
73  FTM_DRV_SetDualEdgeCaptureCmd(ftmBase, chnlPairNum, false);
74  /* Configure the ELSB bit and ELSA bit from user */
75  FTM_DRV_SetChnEdgeLevel(ftmBase, channelId, (uint8_t)param->pwmIndependentChannelConfig[index].levelSelect);
76  /* Set MSB and MSA bits*/
77  FTM_DRV_SetChnMSnBAMode(ftmBase, channelId, 3U);
78  /* Write FTMn_PWMLOAD register to enable synchronized loading points for the given channel */
79  FTM_DRV_EnablePwmChannelOutputs(ftmBase, channelId);
80  FTM_DRV_SetDualChnFaultCmd(ftmBase, chnlPairNum, ((param->faultConfig)->faultMode != FTM_FAULT_CONTROL_DISABLED) ? true : false);
81  /* Enable sync control for channels*/
82  FTM_DRV_SetDualChnPwmSyncCmd(ftmBase, chnlPairNum, true);
83  FTM_DRV_SetDualChnCombineCmd(ftmBase, chnlPairNum, false);
84  FTM_DRV_SetDualChnMofCombineCmd(ftmBase, chnlPairNum, false);
85  /* Enable the generation a trigger on chip module */
86  FTM_DRV_SetChnTriggerCmd(ftmBase, channelId, param->pwmIndependentChannelConfig[index].enableExternalTrigger);
87  /* Enable channel (n+1) output */
88  if( param->pwmIndependentChannelConfig[index].enableSecondChannelOutput)
89  {
90  /* Check if the channel id is even number */
91  DEV_ASSERT((channelId % 2U) == 0U);
92  channelId++;
93  /* Enable output */
94  FTM_DRV_EnablePwmChannelOutputs(ftmBase, channelId);
95  /* Set Edges to enable channel output */
96  FTM_DRV_SetChnEdgeLevel(ftmBase, channelId, (uint8_t)param->pwmIndependentChannelConfig[index].levelSelect);
97  /* Set MSB and MSA bits*/
98  FTM_DRV_SetChnMSnBAMode(ftmBase, channelId, 3U);
99  /* Check if the channel (n+1) is inverse the channel (n) */
100  if (param->pwmIndependentChannelConfig[index].secondChannelPolarity == FTM_MAIN_INVERTED)
101  {
102  FTM_DRV_SetDualChnCompCmd(ftmBase, chnlPairNum, true);
103  }
104  }
105  else
106  {
107  FTM_DRV_SetChnOutputPolarityCmd(ftmBase, channelId, ((param->pwmIndependentChannelConfig[index].polarity == FTM_POLARITY_HIGH)? true : false));
108  }
109 
110  /* Enable dead-time insertion */
111  FTM_DRV_SetDualChnDeadtimeCmd(ftmBase, chnlPairNum, param->pwmIndependentChannelConfig[index].deadTime);
112  }
113 
114  /* Configure combined PWM channels */
115  for (index = 0U; index < param->nNumCombinedPwmChannels; index++)
116  {
117  channelId = param->pwmCombinedChannelConfig[index].hwChannelId;
118  chnlPairNum = (uint8_t)(channelId >> 1U);
119  /* Check if the channel id is even number */
120  DEV_ASSERT((channelId % 2U) == 0U);
121  FTM_DRV_SetDualEdgeCaptureCmd(ftmBase, chnlPairNum, false);
122  /* Configure the ELSB bit and ELSA bit from user */
123  FTM_DRV_SetChnEdgeLevel(ftmBase, channelId, (uint8_t)param->pwmCombinedChannelConfig[index].levelSelect);
124  /* Set MSB and MSA bits */
125  FTM_DRV_SetChnMSnBAMode(ftmBase, channelId, 3U);
126  /* Enable channel (n) output */
127  FTM_DRV_EnablePwmChannelOutputs(ftmBase, channelId);
128  /* Enable channel (n+1) output */
129  if (param->pwmCombinedChannelConfig[index].enableSecondChannelOutput)
130  {
131  FTM_DRV_EnablePwmChannelOutputs(ftmBase, (uint8_t)(channelId + 1U));
132  /* Configure the ELSB bit and ELSA bit from user */
133  FTM_DRV_SetChnEdgeLevel(ftmBase, (uint8_t)(channelId + 1U), (uint8_t)param->pwmCombinedChannelConfig[index].levelSelectOnNextChn);
134  /* Configure complementary mode for channel (n+1) */
135  FTM_DRV_SetDualChnCompCmd(ftmBase, chnlPairNum, ((param->pwmCombinedChannelConfig[index].secondChannelPolarity == FTM_MAIN_INVERTED)? true : false));
136  }
137  else
138  {
139  /* Set polarity for the main channel */
140  FTM_DRV_SetChnOutputPolarityCmd(ftmBase, channelId, ((param->pwmCombinedChannelConfig[index].mainChannelPolarity == FTM_POLARITY_HIGH)? true : false));
141  FTM_DRV_DisablePwmChannelOutputs(ftmBase, (uint8_t)(channelId + 1U));
142  }
143 
144  /* Set fault control for the channel */
145  FTM_DRV_SetDualChnFaultCmd(ftmBase, chnlPairNum, ((param->faultConfig)->faultMode != FTM_FAULT_CONTROL_DISABLED) ? true : false);
146  /* Enable sync control for channels */
147  FTM_DRV_SetDualChnPwmSyncCmd(ftmBase, chnlPairNum, true);
148  /* Enable the combine mode */
149  FTM_DRV_SetDualChnCombineCmd(ftmBase, chnlPairNum, true);
150  /* Configure the modified combine mode */
151  FTM_DRV_SetDualChnMofCombineCmd(ftmBase, chnlPairNum, param->pwmCombinedChannelConfig[index].enableModifiedCombine);
152  /* Configure dead time */
153  FTM_DRV_SetDualChnDeadtimeCmd(ftmBase, chnlPairNum, param->pwmCombinedChannelConfig[index].deadTime);
154  /* Enable the generation a trigger on the channel (n) */
155  FTM_DRV_SetChnTriggerCmd(ftmBase, (chnlPairNum << 1U), param->pwmCombinedChannelConfig[index].enableExternalTrigger);
156  /* Enable the generation a trigger on the channel (n+1) */
157  FTM_DRV_SetChnTriggerCmd(ftmBase, (chnlPairNum << 1U) + 1U, param->pwmCombinedChannelConfig[index].enableExternalTriggerOnNextChn);
158  }
159 
160  /* Set enable outputs to be set to Initial/default value */
161  FTM_DRV_SetInitChnOutputCmd(ftmBase, true);
162  /* Enable faults (if faults were configured) */
163  if ((param->faultConfig)->faultMode != FTM_FAULT_CONTROL_DISABLED)
164  {
165  /* Configure PWM Output behavior */
166  FTM_DRV_SetPwmFaultBehavior(ftmBase, ((param->faultConfig)->pwmOutputStateOnFault) ? true : false);
167  /* Configure fault filter value */
168  FTM_DRV_SetFaultInputFilterVal(ftmBase, ((param->faultConfig)->faultFilterValue));
169  for (fltChannel = 0U; fltChannel < FTM_FEATURE_FAULT_CHANNELS; fltChannel++)
170  {
171  if (true == (param->faultConfig)->ftmFaultChannelParam[fltChannel].faultChannelEnabled)
172  {
173  /* Enable fault channel */
174  FTM_DRV_SetFaultInputCmd(ftmBase, fltChannel, true);
175  /* Configure fault filter */
176  FTM_DRV_SetFaultInputFilterCmd(ftmBase,
177  fltChannel,
178  ((param->faultConfig)->ftmFaultChannelParam[fltChannel].faultFilterEnabled) ? true : false);
179  /* Configure fault outputs */
180  FTM_DRV_SetChnFaultInputPolarityCmd(ftmBase,
181  fltChannel,
182  (((param->faultConfig)->ftmFaultChannelParam[fltChannel].ftmFaultPinPolarity == FTM_POLARITY_HIGH)? true : false));
183  }
184  }
185 
186  /* Set fault interrupt */
187  if (true == ((param->faultConfig)->pwmFaultInterrupt))
188  {
189  FTM_DRV_SetFaultInt(ftmBase, true);
190  }
191 
192  /* Enable fault control */
193  FTM_DRV_SetFaultControlMode(ftmBase, (uint32_t)(param->faultConfig)->faultMode);
194  }
195 
196  /* Configure PWM mode: edge or center aligned */
197  FTM_DRV_SetCpwms(ftmBase, (param->mode == FTM_MODE_CEN_ALIGNED_PWM) ? true : false);
198  /* Calculate frequency of the give FTM hardware module - all channels will run at the same frequency */
199  state->ftmPeriod = FTM_DRV_ConvertFreqToPeriodTicks(instance, param->uFrequencyHZ);
200  /* Based on reference manual, in PWM mode CNTIN is to be set 0*/
201  FTM_DRV_SetCounterInitVal(ftmBase, 0U);
202  /* Write MOD register with the value of the period */
203  /* For center aligned mode MOD register should be divided by 2 */
204  /* For edge aligned mode period is determined by: MOD-CNTIN+1 */
205  if (param->mode == FTM_MODE_CEN_ALIGNED_PWM)
206  {
207  FTM_DRV_SetMod(ftmBase, (uint16_t)(state->ftmPeriod >> 1U));
208  }
209  else
210  {
211  FTM_DRV_SetMod(ftmBase, (uint16_t)(state->ftmPeriod - 1U));
212  }
213 
214  for (index = 0U; index < param->nNumIndependentPwmChannels; index++)
215  {
216  hwChannel = param->pwmIndependentChannelConfig[index].hwChannelId;
217  /* Write CnV registers and setup duty cycle and phase values */
218  retVal = FTM_DRV_UpdatePwmChannel(instance,
219  hwChannel,
220  FTM_PWM_UPDATE_IN_DUTY_CYCLE,
221  param->pwmIndependentChannelConfig[index].uDutyCyclePercent,
222  0U,
223  false);
224  /* Enable channel (n+1) output for the same channel (n) */
225  if ((param->pwmIndependentChannelConfig[index].enableSecondChannelOutput) && (param->pwmIndependentChannelConfig[index].secondChannelPolarity == FTM_MAIN_DUPLICATED))
226  {
227  /* Write CnV registers and setup duty cycle and phase values */
228  retVal = FTM_DRV_UpdatePwmChannel(instance,
229  hwChannel + 1U,
230  FTM_PWM_UPDATE_IN_DUTY_CYCLE,
231  param->pwmIndependentChannelConfig[index].uDutyCyclePercent,
232  0U,
233  false);
234  }
235  }
236 
237  for (index = 0U; index < param->nNumCombinedPwmChannels; index++)
238  {
239  hwChannel = param->pwmCombinedChannelConfig[index].hwChannelId;
240  /* Write CnV registers and setup duty cycle and phase values */
241  retVal = FTM_DRV_UpdatePwmChannel(instance,
242  hwChannel,
243  FTM_PWM_UPDATE_IN_DUTY_CYCLE,
244  param->pwmCombinedChannelConfig[index].firstEdge,
245  param->pwmCombinedChannelConfig[index].secondEdge,
246  false);
247  }
248 
249  if (STATUS_SUCCESS == retVal)
250  {
251  /* Configure dead time for combine mode */
252  FTM_DRV_SetDeadtimeCount(ftmBase, param->deadTimeValue);
253  FTM_DRV_SetDeadtimePrescale(ftmBase, param->deadTimePrescaler);
254  FTM_DRV_Enable(ftmBase, true);
255  FTM_DRV_SetPwmSyncMode(ftmBase, true);
256  /* Set clock source to start counter */
257  FTM_DRV_SetClockSource(ftmBase, state->ftmClockSource);
258  }
259  else
260  {
261  /* Restore FTM mode if initialization fails */
262  state->ftmMode = FTM_MODE_NOT_INITIALIZED;
263  }
264  }
265  else
266  {
267  retVal = STATUS_ERROR;
268  }
269 
270  return retVal;
271 }

273 /*FUNCTION**********************************************************************
274  *
275  * Function Name : FTM_DRV_DeinitPwm
276  * Description : Stops all PWM channels.
277  *
278  * Implements : FTM_DRV_DeinitPwm_Activity
279  *END**************************************************************************/
280 status_t FTM_DRV_DeinitPwm(uint32_t instance)
281 {
282  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
283  FTM_Type * ftmBase = g_ftmBase[instance];
284  uint8_t channel;
285  uint8_t chnlPairNum;
286  ftm_state_t * state = ftmStatePtr[instance];
287 
288  /* Stop the FTM counter */
289  FTM_DRV_SetClockSource(ftmBase, FTM_CLOCK_SOURCE_NONE);
290  for (channel = 0U; channel < FEATURE_FTM_CHANNEL_COUNT; channel++)
291  {
292  chnlPairNum = (uint8_t)(channel >> 1U);
293  /* Disable PWM mode in hardware */
294  FTM_DRV_SetChnCountVal(ftmBase, channel, 0U);
295  FTM_DRV_SetChnEdgeLevel(ftmBase, channel, 0U);
296  FTM_DRV_SetChnMSnBAMode(ftmBase, channel, 0U);
297  FTM_DRV_SetCpwms(ftmBase, false);
298  /* Configure polarity bit */
299  FTM_DRV_SetChnOutputPolarityCmd(ftmBase, channel, false);
300  FTM_DRV_DisablePwmChannelOutputs(ftmBase, channel);
301  /* Clear the PWM synchronization */
302  FTM_DRV_SetDualChnPwmSyncCmd(ftmBase, chnlPairNum, false);
303  /* Clear combination for each pair of channels */
304  FTM_DRV_SetDualChnMofCombineCmd(ftmBase, chnlPairNum, false);
305  FTM_DRV_SetDualChnCombineCmd(ftmBase, chnlPairNum, false);
306  FTM_DRV_SetDualChnDeadtimeCmd(ftmBase, chnlPairNum, false);
307  FTM_DRV_SetDualChnFaultCmd(ftmBase, chnlPairNum, false);
308  }
309 
310  /* Clear the dead-time pre-scaler and value */
311  FTM_DRV_SetExtDeadtimeValue(ftmBase, 0U);
312  FTM_DRV_SetDeadtimePrescale(ftmBase, FTM_DEADTIME_DIVID_BY_1);
313  FTM_DRV_SetDeadtimeCount(ftmBase, 0U);
314  /* Clear fault control register */
315  FTM_DRV_ClearFaultControl(ftmBase);
316  /* Disable fault interrupt */
317  FTM_DRV_SetFaultInt(ftmBase, false);
318  /* Disable fault control */
319  FTM_DRV_SetFaultControlMode(ftmBase, (uint32_t)FTM_FAULT_CONTROL_DISABLED);
320  /* Clear the module value of the registers */
321  FTM_DRV_SetMod(ftmBase, 0U);
322  FTM_DRV_SetCounter(ftmBase, 0U);
323  state->ftmMode = FTM_MODE_NOT_INITIALIZED;
324 
325  return STATUS_SUCCESS;
326 }

328 /*FUNCTION**********************************************************************
329  *
330  * Function Name : FTM_DRV_UpdatePwmChannel
331  * Description : This function will update the duty cycle of PWM output.
332  * - If the type of update in the duty cycle, this function will convert the input parameters representing
333  * frequency in Hz to a period value in ticks needed by the hardware timer. Period is calculated depending
334  * on the operating mode of the FTM module and is stored in internal state structure.
335  * firstEdge and secondEdge can have value between 0 - FTM_MAX_DUTY_CYCLE(0 = 0% duty
336  * and FTM_MAX_DUTY_CYCLE = 100% duty). secondEdge value is used only whenFTM module is used in PWM combine mode.
337  * - If the type of update in ticks, this function will get value in ticks to the hardware timer.
338  * firstEdge and secondEdge variables can have value between 0 and ftmPeriod is stored in the state structure.
339  * - in the modified combine mode, the firstEdge parameter is fixed value and only can modify the secondEdge variables
340  * which is the initial value in the channel (n+1) match edge when the FTM counter has been ran.
341  *
342  * Implements : FTM_DRV_UpdatePwmChannel_Activity
343  *END**************************************************************************/
344 status_t FTM_DRV_UpdatePwmChannel(uint32_t instance,
345  uint8_t channel,
346  ftm_pwm_update_option_t typeOfUpdate,
347  uint16_t firstEdge,
348  uint16_t secondEdge,
349  bool softwareTrigger)
350 {
351  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
352  DEV_ASSERT(channel < FEATURE_FTM_CHANNEL_COUNT);
353  uint16_t hwFirstEdge = 0U;
354  uint16_t hwSecondEdge = 0U;
355  uint16_t ftmPeriod = 0U;
356  uint8_t chnlPairNum = (uint8_t)(channel >> 1U);
357  FTM_Type * ftmBase = g_ftmBase[instance];
358  ftm_state_t * state = ftmStatePtr[instance];
359  status_t retStatus = STATUS_SUCCESS;
360 
361  /* Get the newest period in the MOD register */
362  ftmPeriod = FTM_DRV_GetMod(ftmBase);
363  /* Check the mode operation in FTM module */
364  if (state->ftmMode == FTM_MODE_CEN_ALIGNED_PWM)
365  {
366  ftmPeriod = (uint16_t)(ftmPeriod << 1U);
367  }
368  else if (state->ftmMode == FTM_MODE_EDGE_ALIGNED_PWM)
369  {
370  ftmPeriod = (uint16_t)(ftmPeriod + 1U);
371  }
372  else
373  {
374  retStatus = STATUS_ERROR;
375  }
376 
377  /* Check the type of update for PWM */
378  if (FTM_PWM_UPDATE_IN_DUTY_CYCLE == typeOfUpdate)
379  {
380  if ((firstEdge <= FTM_MAX_DUTY_CYCLE) && (secondEdge <= FTM_MAX_DUTY_CYCLE))
381  {
382  /* Calculate DutyCycle based of the previously calculated frequency*/
383  /* For greater resolution the DutyCycle values are in the range [0. FTM_MAX_DUTY_CYCLE]
384  * where 0 = 0% or PWM always at Low and FTM_MAX_DUTY_CYCLE = 100% or PWM always HIGH;
385  * a value of 0x4000 is equivalent of 50% DutyCycle. */
386  hwFirstEdge = (uint16_t)((ftmPeriod * firstEdge) >> FTM_DUTY_TO_TICKS_SHIFT);
387  hwSecondEdge = (uint16_t)((ftmPeriod * secondEdge) >> FTM_DUTY_TO_TICKS_SHIFT);
388  /* adjust DutyCycle if 100% value is to be achieved. */
389  if (FTM_MAX_DUTY_CYCLE == firstEdge)
390  {
391  /* If expected duty is 100% then increase by 1 the value that is to be written
392  * to Hardware so it will exceed value of period */
393  hwFirstEdge = (uint16_t)(hwFirstEdge + 1U);
394  }
395  }
396  else
397  {
398  retStatus = STATUS_ERROR;
399  }
400  }
401  else
402  {
403  if ((firstEdge <= ftmPeriod) && (secondEdge <= ftmPeriod))
404  {
405  hwFirstEdge = firstEdge;
406  hwSecondEdge = secondEdge;
407  }
408  else
409  {
410  retStatus = STATUS_ERROR;
411  }
412  }
413 
414  if (STATUS_SUCCESS == retStatus)
415  {
416  if (true == FTM_DRV_GetDualChnCombineCmd(ftmBase, chnlPairNum))
417  {
418  if (true == FTM_DRV_GetDualChnMofCombineCmd(ftmBase, chnlPairNum))
419  {
420  /* Check the clock source for FTM counter is disabled or not */
421  if (FTM_DRV_GetClockSource(ftmBase) == 0U)
422  {
423  FTM_DRV_SetChnCountVal(ftmBase, (uint8_t)(chnlPairNum * 2U), hwFirstEdge);
424  }
425  }
426  else
427  {
428  FTM_DRV_SetChnCountVal(ftmBase, (uint8_t)(chnlPairNum * 2U), hwFirstEdge);
429  }
430 
431  /* Modify the initial value in the channel (n+1) match edge */
432  FTM_DRV_SetChnCountVal(ftmBase, (uint8_t)((chnlPairNum * 2U) + 1U), hwSecondEdge);
433  }
434  else
435  {
436  /* Channel value is divided by 2 for up down counter mode to keep same duty */
437  if (true == FTM_DRV_GetCpwms(ftmBase))
438  {
439  FTM_DRV_SetChnCountVal(ftmBase, channel, (uint16_t)(hwFirstEdge >> 1U));
440  }
441  else
442  {
443  FTM_DRV_SetChnCountVal(ftmBase, channel, hwFirstEdge);
444  }
445  }
446 
447  /* Software trigger is generated to change CnV registers */
448  /* Before this please configure sync mechanism to use software trigger */
449  FTM_DRV_SetSoftwareTriggerCmd(ftmBase, softwareTrigger);
450 
451  /* Store the PWM period in the state structure */
452  state->ftmPeriod = ftmPeriod;
453  }
454 
455  return retStatus;
456 }

458 /*FUNCTION**********************************************************************
459  *
460  * Function Name : FTM_DRV_UpdatePwmPeriod
461  * Description : This function will update the new period in the frequency or
462  * in the counter value into mode register which modify the period of PWM signal
463  * on the channel output.
464  * - If the type of update in the duty cycle which is reused in FTM_DRV_UpdatePwmChannel
465  * function to convert the newValue parameters representing frequency in Hz to
466  * a period value to update the MOD register. The newValue parameter must be value
467  * between 1U and maximum is the frequency of the FTM counter.
468  * - If the type of update in ticks, this function will get value in counting to
469  * the MOD register. The newValue parameter must be value between 1U and 0xFFFFU
470  *
471  * Implements : FTM_DRV_UpdatePwmPeriod_Activity
472  *END**************************************************************************/
473 status_t FTM_DRV_UpdatePwmPeriod(uint32_t instance,
474  ftm_pwm_update_option_t typeOfUpdate,
475  uint32_t newValue,
476  bool softwareTrigger)
477 {
478  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);
479  DEV_ASSERT(newValue != 0U);
480  uint32_t ftmPeriod = 0U;
481  FTM_Type * ftmBase = g_ftmBase[instance];
482  ftm_state_t * state = ftmStatePtr[instance];
483  status_t retStatus = STATUS_SUCCESS;
484 
485  /* Check the type of update for period in PWM mode */
486  if (FTM_PWM_UPDATE_IN_TICKS == typeOfUpdate)
487  {
488  ftmPeriod = newValue;
489  }
490  else
491  {
492  if (newValue <= state->ftmSourceClockFrequency)
493  {
494  ftmPeriod = (uint32_t)FTM_DRV_ConvertFreqToPeriodTicks(instance, newValue);
495  }
496  else
497  {
498  retStatus = STATUS_ERROR;
499  }
500  }
501 
502  if (STATUS_SUCCESS == retStatus)
503  {
504  /* Check the ftmPeriod is invalid */
505  DEV_ASSERT(ftmPeriod <= 0xFFFFU);
506  /* Check the signal operation in which PWM mode */
507  DEV_ASSERT((FTM_MODE_CEN_ALIGNED_PWM == state->ftmMode) || (FTM_MODE_EDGE_ALIGNED_PWM == state->ftmMode));
508  if (FTM_MODE_CEN_ALIGNED_PWM == state->ftmMode)
509  {
510  ftmPeriod = (ftmPeriod >> 1U);
511  }
512  else
513  {
514  ftmPeriod = (ftmPeriod - 1U);
515  }
516  /* Set the new modulo value into MOD register */
517  FTM_DRV_SetMod(ftmBase, (uint16_t)(ftmPeriod));
518  /* Software trigger is generated to change MOD registers */
519  FTM_DRV_SetSoftwareTriggerCmd(ftmBase, softwareTrigger);
520 
521  /* Store the PWM period in the state structure */
522  state->ftmPeriod = (uint16_t)ftmPeriod;
523  }
524 
525  return retStatus;
526 }

528 /*FUNCTION**********************************************************************
529  *
530  * Function Name : FTM_DRV_FastUpdatePwmChannels
531  * Description : This function will update the duty cycle of PWM output for multiple channels.
532  *
533  * The main differences between this function and FTM_DRV_UpdatePwmChannel is the execution speed. This

620  * Function Name : FTM_PWM_DRV_IrqHandler
621  * Description : This function is used to workaround an errata which the safe state
622  * is not removed from channel outputs after fault condition ends if SWOCTRL is being
623  * used to control the pin. The FTM_MODE[FAULTM] should be configured for manual fault
624  * clearing (0b10)
625  * This function must be used in the TOF interrupt handler when a fault is detected to
626  * ensure that the outputs return to the value configured by FTM_SWOCTR
627  *
628  * Implements : FTM_PWM_DRV_IrqHandler_Activity
629  *END**************************************************************************/
630 void FTM_PWM_DRV_IrqHandler(uint32_t instance,
631  uint32_t chnOutCtrlVal)
632 {
633  FTM_Type * ftmBase = g_ftmBase[instance];
634 
635  /* 1. Check the value of FTM_FMS[FAULTF] */
636  if (FTM_DRV_GetDetectedFaultInput(ftmBase) == true)
637  {
638  faultDetection = true;
639  /* 2. Write the FTM_OUTMASK register to set the bit */
640  FTM_DRV_SetOutmaskReg(ftmBase, chnOutCtrlVal);
641  /* 3. Clear fault conditions by reading the FTM_FMS register and then writing FTM_FMS with all zeroes */
642  FTM_DRV_ClearFaultsIsr(ftmBase);
643  /* 4. Clear the FTM_SC[TOF] bit by reading the FTM_SC register, then writing a 0 to FTM_SC[TOF] */
644  FTM_DRV_ClearTimerOverflow(ftmBase);
645  /* 5. Exit the interrupt handler to skip following steps */
646  }
647  else if (faultDetection == true)
648  {
649  /* 6. Clear the FTM_SWOCTRL by writing all zeroes to it */
650  FTM_DRV_SetAllChnSoftwareCtrlCmd(ftmBase, 0x00U);
651  FTM_DRV_SetAllChnSoftwareCtrlVal(ftmBase, 0x00U);
652  /* 7. Write FTM_SWOCTRL with the desired value again */
653  uint8_t u8chnOutCtrlVal = (uint8_t)(chnOutCtrlVal & 0xFFu);
654  FTM_DRV_SetAllChnSoftwareCtrlCmd(ftmBase, u8chnOutCtrlVal);
655  FTM_DRV_SetAllChnSoftwareCtrlVal(ftmBase, u8chnOutCtrlVal);
656  /* 8. Clear the FTM_OUTMASK bits that were set in step 2 */
657  FTM_DRV_SetOutmaskReg(ftmBase, 0x00U);
658  /* 9. Clear the fault variable that was set in step 1 when the fault condition was originally detected */
659  faultDetection = false;
660  /* 10. Clear the FTM_SC[TOF] bit by reading the FTM_SC register, then writing a 0 to FTM_SC[TOF] */
661  FTM_DRV_ClearTimerOverflow(ftmBase);
662  }
663  else
664  {
665  /* Nothing to do */
666  }
667 }
