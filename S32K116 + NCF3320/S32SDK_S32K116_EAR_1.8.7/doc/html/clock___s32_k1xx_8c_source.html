465 /*FUNCTION**********************************************************************
466  *
467  * Function Name : CLOCK_SYS_SetConfiguration
468  * Description : This function sets the system to target configuration, it
469  * only sets the clock modules registers for clock mode change, but not send
470  * notifications to drivers.
471  *
472  * Implements CLOCK_SYS_SetConfiguration_Activity
473  * END**************************************************************************/
474 status_t CLOCK_SYS_SetConfiguration(clock_manager_user_config_t const * config)
475 {
476  status_t result;
477  clock_manager_user_config_t config_default;
478  clock_manager_user_config_t const * cfg = config;
479 
480  if (config == NULL)
481  {
482  /* Get default configuration */
483  CLOCK_SYS_GetDefaultConfiguration(&config_default);
484  cfg = &config_default;
485  }
486 
487  /* Set SCG settings. */
488  result = CLOCK_SYS_SetScgConfiguration(&cfg->scgConfig);
489 
490  if (STATUS_SUCCESS == result)
491  {
492  /* Set PCC settings. */
493  CLOCK_SYS_SetPccConfiguration(&cfg->pccConfig);
494 
495  /* Set SIM settings. */
496  CLOCK_SYS_SetSimConfiguration(&cfg->simConfig);
497 
498  /* Set PMC settings. */
499  CLOCK_SYS_SetPmcConfiguration(&cfg->pmcConfig);
500  }
501 
502  return result;
503 }

505 /*FUNCTION**********************************************************************
506  *
507  * Function Name : CLOCK_SYS_SetScgConfiguration
508  * Description : This function configures the SCG blocks
509  *
510  * Implements CLOCK_SYS_SetScgConfiguration_Activity
511  * END**************************************************************************/
512 static status_t CLOCK_SYS_SetScgConfiguration(const scg_config_t * scgConfig)
513 {
514  status_t status = STATUS_SUCCESS;
515  DEV_ASSERT(scgConfig != NULL);
516 
517  if (scgConfig != NULL)
518  {
519  /* Configure a temporary system clock source: FIRC */
520  status = CLOCK_SYS_ConfigureTemporarySystemClock();
521 
522  if (status == STATUS_SUCCESS)
523  {
524  /* Configure clock sources from SCG */
525  status = CLOCK_SYS_ConfigureModulesFromScg(scgConfig);
526  }
527 
528  if (status == STATUS_SUCCESS)
529  {
530  /* Configure RTC. */
531  if (scgConfig->rtcConfig.initialize)
532  {
533  /* RTC Clock settings. */
534  g_RtcClkInFreq = scgConfig->rtcConfig.rtcClkInFreq;
535  }
536 
537  /* Configure SCG ClockOut. */
538  if (scgConfig->clockOutConfig.initialize)
539  {
540  /* ClockOut settings. */
541  SCG_SetClockoutSourceSel(SCG, (uint32_t)scgConfig->clockOutConfig.source);
542  }
543 
544  /* Configure SCG clock modes. */
545  if (scgConfig->clockModeConfig.initialize)
546  {
547  /* Configure SCG clock modes */
548  status = CLOCK_SYS_SetSystemClockConfig(SCG_SYSTEM_CLOCK_MODE_RUN, &(scgConfig->clockModeConfig.rccrConfig));
549  if (status == STATUS_SUCCESS)
550  {
551  status = CLOCK_SYS_SetSystemClockConfig(SCG_SYSTEM_CLOCK_MODE_VLPR, &(scgConfig->clockModeConfig.vccrConfig));
552  }
553 
554 #if FEATURE_HAS_HIGH_SPEED_RUN_MODE
555  if (status == STATUS_SUCCESS)
556  {
557  status = CLOCK_SYS_SetSystemClockConfig(SCG_SYSTEM_CLOCK_MODE_HSRUN, &(scgConfig->clockModeConfig.hccrConfig));
558  }
559 #endif
560  }
561  }
562  }
563 
564  return status;
565 }

637  }
638

641  {

643  {
644  if (i < NUMBER_OF_TCLK_INPUTS)
645  {

647  }
648  }
649  }
650 
651  /* Debug trace Clock settings. */
652  if (simClockConfig->traceClockConfig.initialize)
653  {
654  SIM_ClearTraceClockConfig(SIM);
655  SIM_SetTraceClockSource(SIM, simClockConfig->traceClockConfig.source);
656 
657  SIM_SetTraceClockConfig(SIM, false, 0U, 0U);
658 
659  if (simClockConfig->traceClockConfig.divEnable)
660  {
661  SIM_SetTraceClockConfig(SIM,
662  simClockConfig->traceClockConfig.divEnable,
663  simClockConfig->traceClockConfig.divider,
664  simClockConfig->traceClockConfig.divFraction);
665  }
666  }
