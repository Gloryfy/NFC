79  * Description : Initialize the WDG PAL
80  *
81  * Implements : WDG_Init_Activity
82  *END**************************************************************************/
83 status_t WDG_Init(wdg_instance_t instance,
84  const wdg_config_t * configPtr)
85 {
86  DEV_ASSERT(configPtr != NULL);
87 
88  uint32_t windowValue = 0U;
89  status_t status = STATUS_ERROR;
90  uint32_t valInstance = (uint32_t)instance;
91 
92  /* Calculate window value */
93  if ((configPtr->winEnable) && (configPtr->percentWindow <= 100U))
94  {
95  windowValue = (uint32_t)(((uint64_t)configPtr->percentWindow * configPtr->timeoutValue) / 100U);
96  }
97 
98  /* Define WDG PAL over WDOG */
99 #if defined (WDG_OVER_WDOG)
100  if (valInstance <= WDG_OVER_WDOG_HIGH_INDEX)
101  {
102  /* Define structures for WDG PAL over WDOG */
103  wdog_user_config_t wdogConfig =
104  {
105  .clkSource = (wdog_clk_source_t)configPtr->clkSource,
106  .opMode =
107  {
108  .wait = configPtr->opMode.wait,
109  .stop = configPtr->opMode.stop,
110  .debug = configPtr->opMode.debug
111  },
112  .updateEnable = true,
113  .intEnable = configPtr->intEnable,
114  .winEnable = configPtr->winEnable,
115  .windowValue = (uint16_t)((configPtr->timeoutValue - windowValue) & 0xFFFFU),
116  .timeoutValue = (uint16_t)(configPtr->timeoutValue & 0xFFFFU),
117  .prescalerEnable = configPtr->prescalerEnable
118  };
119 
120  /* Initialize WDOG instance */
121  status = WDOG_DRV_Init(valInstance, &wdogConfig);
122  }
123 #endif
124 
125  /* Define WDG PAL over EWM */
126 #if defined (WDG_OVER_EWM)
127  if ((valInstance >= WDG_OVER_EWM_LOW_INDEX) && (valInstance <= WDG_OVER_EWM_HIGH_INDEX))
128  {
129  /* Define structures for WDG PAL over EWM */
130  ewm_init_config_t ewmConfig;
131 
132  /* Set value for EWM structure */
133  ewmConfig.interruptEnable = configPtr->intEnable;
134 
135  /* Set value for input pin asserts and clock prescaler */
136  if (NULL == configPtr->extension)
137  {
138  ewmConfig.assertLogic = EWM_IN_ASSERT_ON_LOGIC_ZERO;
139  ewmConfig.prescaler = 0xFFU;
140  }
141  else
142  {
143  ewmConfig.assertLogic = (ewm_in_assert_logic_t)((extension_ewm_for_wdg_t*)(configPtr->extension))->assertLogic;
144  ewmConfig.prescaler = ((extension_ewm_for_wdg_t*)(configPtr->extension))->prescalerValue;
145  }
146 
147  /* Set value for compare low register */
148  if (true == configPtr->winEnable)
149  {
150  ewmConfig.compareLow = (uint8_t)((configPtr->timeoutValue - windowValue) & 0xFFU);
151  }
152  else
153  {
154  ewmConfig.compareLow = FEATURE_EWM_CMPL_MIN_VALUE;
155  }
156 
157  /* Set value for compare high register */
158  if (configPtr->timeoutValue > FEATURE_EWM_CMPH_MAX_VALUE)
159  {
160  ewmConfig.compareHigh = FEATURE_EWM_CMPH_MAX_VALUE;
161  }
162  else
163  {
164  ewmConfig.compareHigh = (uint8_t)(configPtr->timeoutValue & 0xFFU);
165  }
166 
167  /* Allocate instance for EWM */
168  valInstance -= WDG_OVER_EWM_LOW_INDEX;
169 
170  /* Initialize EWM instance */
171  status = EWM_DRV_Init(valInstance, &ewmConfig);
172  }
173 #endif
174 
175  /* Define WDG PAL over SWT */
176 #if defined (WDG_OVER_SWT)
177  if (valInstance <= WDG_OVER_SWT_HIGH_INDEX)
178  {
179  /* Define structures for WDG PAL over SWT */
180  swt_user_config_t swtConfig =
181  {
182  .mapConfig = 0xFFU,
183  .invalidReset = false,
184  .winEnable = configPtr->winEnable,
185  .intEnable = configPtr->intEnable,
186  .stop = configPtr->opMode.stop,
187  .debug = configPtr->opMode.debug,
188  .serviceMode = SWT_FS_SEQ_MODE,
189  .lockConfig = SWT_UNLOCK,
190  .timeoutValue = configPtr->timeoutValue,
191  .windowValue = windowValue,
192  .initKey = 0U
193  };
194 
195  /* Initialize SWT instance */
196  status = SWT_DRV_Init(valInstance, &swtConfig);
197  }
198 
199 #endif
200 
201  return status;
202 }

301  /* Allocate instance for EWM */
302  valInstance -= WDG_OVER_EWM_LOW_INDEX;
303  /* Refresh counter over EWM module */

305  }
306 #endif
307 
308  /* Define WDG PAL over SWT */
309 #if defined(WDG_OVER_SWT)
310  if (valInstance <= WDG_OVER_SWT_HIGH_INDEX)
311  {
312  /* Refresh counter over SWT module */

423 /*FUNCTION**********************************************************************
424  *
425  * Function Name : WDG_GetCounter
426  * Description : Gets the value of the WDG PAL counter.
427  * Note that: Counter will be reset to timeout value if WDG PAL uses SWT.
428  * The counter will continue to run if WDG PAL uses WDOG.
429  *
430  * Implements : WDG_GetCounter_Activity
431  *END**************************************************************************/
432 status_t WDG_GetCounter(wdg_instance_t instance,
433  uint32_t * value)
434 {
435  uint32_t valInstance = (uint32_t)instance;
436  status_t status = STATUS_UNSUPPORTED;
437 
438  /* Define WDG PAL over WDOG */
439 #if defined(WDG_OVER_WDOG)
440  if (valInstance <= WDG_OVER_WDOG_HIGH_INDEX)
441  {
442  /* Get counter value over WDOG module */
443  *value = WDOG_DRV_GetCounter(valInstance);
444  status = STATUS_SUCCESS;
445  }
446 #endif
447 
448  /* Define WDG PAL over SWT */
449 #if defined(WDG_OVER_SWT)
450  if (valInstance <= WDG_OVER_SWT_HIGH_INDEX)
451  {
452  /* Disable SWT */
453  status = SWT_DRV_StopTimer(valInstance);
454  /* Get counter value of SWT */
455  if (STATUS_SUCCESS == status)
456  {
457  /* Get counter value over SWT module */
458  (void)SWT_DRV_GetCounterValue(valInstance, value);
459  /* Enable SWT */
460  (void)SWT_DRV_StartTimer(valInstance);
461  }
462  }
463 #endif
464 
465  return status;
466 }
