151 /*FUNCTION**********************************************************************
152  *
153  * Function Name : ADC_DRV_Reset
154  * Description : This function writes all the internal ADC registers with
155  * their Reference Manual reset values.
156  *
157  * Implements : ADC_DRV_Reset_Activity
158  *END**************************************************************************/
159 void ADC_DRV_Reset(const uint32_t instance)
160 {
161  DEV_ASSERT(instance < ADC_INSTANCE_COUNT);
162 
163  ADC_Type * const baseAddr = s_adcBase[instance];
164  uint8_t idx = 0U;
165 
166  for(idx = 0U; idx < ADC_SC1_COUNT; idx++)
167  {
168  baseAddr->SC1[idx] = ADC_SC1_ADCH(ADC_INPUTCHAN_DISABLED) | ADC_SC1_AIEN(0x00U);
169  }
170 
171  baseAddr->CFG1 = ADC_CFG1_ADICLK(ADC_CLK_ALT_1) | ADC_CFG1_MODE(ADC_RESOLUTION_8BIT) | ADC_CFG1_ADIV(ADC_CLK_DIVIDE_1);
172  baseAddr->CFG2 = ADC_CFG2_SMPLTS(ADC_DEFAULT_SAMPLE_TIME);
173 
174  for(idx = 0U; idx < ADC_CV_COUNT; idx++)
175  {
176  baseAddr->CV[idx] = ADC_CV_CV(0U);
177  }
178 
179  baseAddr->SC2 = ADC_SC2_REFSEL(ADC_VOLTAGEREF_VREF) | ADC_SC2_DMAEN(0x00U) | ADC_SC2_ACREN(0x00U) | ADC_SC2_ACFGT(0x00U) | ADC_SC2_ACFE(0x00U) |
180  ADC_SC2_ADTRG(0x00U);
181  baseAddr->SC3 = ADC_SC3_AVGS(ADC_AVERAGE_4) | ADC_SC3_AVGE(0x00U) | ADC_SC3_ADCO(0x00U) | ADC_SC3_CAL(0x00U);
182  baseAddr->USR_OFS = ADC_USR_OFS_USR_OFS(0U);
183  baseAddr->UG = ADC_UG_UG(ADC_DEFAULT_USER_GAIN);
184 
185 #if FEATURE_ADC_HAS_EXTRA_NUM_REGS
186  for(idx = 0U; idx < ADC_aSC1_COUNT; idx++)
187  {
188  baseAddr->aSC1[idx] = ADC_aSC1_ADCH(ADC_INPUTCHAN_DISABLED) | ADC_aSC1_AIEN(0x00U);
189  }
190 #endif /* FEATURE_ADC_HAS_EXTRA_NUM_REGS */
191 
192  ADC_SetPretriggerSelect(instance, ADC_PRETRIGGER_SEL_PDB);
193  ADC_SetTriggerSelect(instance, ADC_TRIGGER_SEL_PDB);
194  ADC_DRV_SetSwPretrigger(instance, ADC_SW_PRETRIGGER_DISABLED);
195 
196  /* Reset ADC Supply Monitoring - available only for ADC 0 */
197  if(instance == 0u)
198  {
199  SIM_Type * const simBase = SIM;
200  ADC_SetSupplyMonitoringEnableFlag(simBase, false);
201 
202  simBase->CHIPCTL &= ~SIM_CHIPCTL_ADC_SUPPLY_MASK;
203  }
204 }

348 /*FUNCTION**********************************************************************
349  *
350  * Function Name : ADC_DRV_ConfigChan
351  * Description : This function sets a control channel configuration.
352  *
353  * When Software Trigger mode is enabled, configuring control channel index 0,
354  * implicitly triggers a new conversion on the selected ADC input channel.
355  * Therefore, ADC_DRV_ConfigChan can be used for sw-triggering conversions.
356  *
357  * Configuring any control channel while it is actively controlling a conversion
358  * (sw or hw triggered) will implicitly abort the on-going conversion.
359  *
360  * Implements : ADC_DRV_ConfigChan_Activity
361  *END**************************************************************************/
362 void ADC_DRV_ConfigChan(const uint32_t instance,
363  const uint8_t chanIndex,
364  const adc_chan_config_t * const config)
365 {
366  DEV_ASSERT(instance < ADC_INSTANCE_COUNT);
367  DEV_ASSERT(chanIndex < ADC_CTRL_CHANS_COUNT);
368  DEV_ASSERT(config != NULL);
369 
370  ADC_Type * const base = s_adcBase[instance];
371 
372  /* ADC_INPUTCHAN_SUPPLY_ can only be used with ADC 0,
373  * If used, the feature must be enabled separately via supplyMonitoringEnable flag in adc_converter_config_t. */
374  DEV_ASSERT((instance == 0u) || ((uint32_t)config->channel < (uint32_t)ADC_INPUTCHAN_SUPPLY_VDD) || \
375  ((uint32_t)config->channel > (uint32_t)ADC_INPUTCHAN_SUPPLY_VDD_LV));
376  ADC_SetInputChannel(base, chanIndex, config->channel, config->interruptEnable);
377 }

400 /*FUNCTION**********************************************************************
401  *
402  * Function Name : ADC_DRV_SetSwPretrigger
403  * Description : This function sets the software pretrigger - affects only first 4 control channels.
404  *
405  * Implements : ADC_DRV_SetSwPretrigger_Activity
406  *END**************************************************************************/
407 void ADC_DRV_SetSwPretrigger(const uint32_t instance,
408  const adc_sw_pretrigger_t swPretrigger)
409 {
410  DEV_ASSERT(instance < ADC_INSTANCE_COUNT);
411  SIM_Type * const simBase = SIM;
412  uint32_t intermValue = 0U;
413 #if (ADC_INSTANCE_COUNT == 1U)
414  const uint32_t mask[ADC_INSTANCE_COUNT] = {SIM_ADCOPT_ADC0SWPRETRG_MASK};
415 #elif (ADC_INSTANCE_COUNT == 2U)
416  const uint32_t mask[ADC_INSTANCE_COUNT] = {SIM_ADCOPT_ADC0SWPRETRG_MASK, SIM_ADCOPT_ADC1SWPRETRG_MASK};
417 #endif
418  /* If SW Pretrigger Select is not enabled, the SW pretriggers will be ignored by ADC. */
419  DEV_ASSERT((ADC_GetPretriggerSelect(instance) == ADC_PRETRIGGER_SEL_SW) || \
420  (swPretrigger == ADC_SW_PRETRIGGER_DISABLED));
421 
422  intermValue = simBase->ADCOPT & (~ mask[instance]);
423  switch(instance)
424  {
425  case 0:
426  intermValue |= SIM_ADCOPT_ADC0SWPRETRG(swPretrigger);
427  break;
428  case 1:
429  intermValue |= SIM_ADCOPT_ADC1SWPRETRG(swPretrigger);
430  break;
431  default:
432  DEV_ASSERT(false);
433  break;
434  }
435 
436  simBase->ADCOPT = intermValue;
437 }

458 /*FUNCTION**********************************************************************
459  *
460  * Function Name : ADC_DRV_GetConvCompleteFlag
461  * Description : This function returns the state of the Conversion Complete
462  * flag for a control channel. This flag is set when a conversion is complete
463  * or the condition generated by the Hardware Compare feature is evaluated to true.
464  *
465  * Implements : ADC_DRV_GetConvCompleteFlag_Activity

468  const uint8_t chanIndex)
469 {

520 /*FUNCTION**********************************************************************
521  *
522  * Function Name : ADC_DRV_AutoCalibration
523  * Description : This functions executes an Auto-Calibration sequence. It
524  * is recommended to run this sequence before using the ADC converter.
525  *
526  * Implements : ADC_DRV_AutoCalibration_Activity
527  *END**************************************************************************/
528 void ADC_DRV_AutoCalibration(const uint32_t instance)
529 {
530  DEV_ASSERT(instance < ADC_INSTANCE_COUNT);
531 
532  ADC_Type * const base = s_adcBase[instance];
533  /* set hardware average to maximum and set software trigger*/
534  bool hwavgen = ADC_GetHwAverageEnableFlag(base);
535  adc_average_t hwavg = ADC_GetHwAverageMode(base);
536  adc_trigger_t trig = ADC_GetTriggerMode(base);
537  ADC_SetHwAverageMode(base, ADC_AVERAGE_32);
538  ADC_SetHwAverageEnableFlag(base, true);
539  ADC_SetTriggerMode(base, ADC_TRIGGER_SOFTWARE);
540 
541  base->CLPS = 0x00u;
542  base->CLP3 = 0x00u;
543  base->CLP2 = 0x00u;
544  base->CLP1 = 0x00u;
545  base->CLP0 = 0x00u;
546  base->CLPX = 0x00u;
547  base->CLP9 = 0x00u;
548 
549  /* start calibration */
550  ADC_SetCalibrationActiveFlag(base, true);
551  while (ADC_GetCalibrationActiveFlag(base))
552  {
553  /* Wait for calibration to finish */
554  }
555 
556  /* restore hardware average and trigger settings*/
557  ADC_SetHwAverageEnableFlag(base, hwavgen);
558  ADC_SetHwAverageMode(base, hwavg);
559  ADC_SetTriggerMode(base, trig);
560 }
