116  * Description : Allocates one of the available state structure.
117  *
118  *END**************************************************************************/
119 static uint8_t SpiAllocateState(bool* isAllocated, spi_instance_t* instanceMapping, spi_instance_t instance, uint8_t numberOfinstances)
120 {
121  uint8_t i;
122  /* Allocate one of the LPSPI state structure for this instance */
123  for (i = 0;i < numberOfinstances;i++)
124  {
125  if (isAllocated[i] == false)
126  {
127  instanceMapping[i] = instance;
128  isAllocated[i] = true;
129  break;
130  }
131  }
132  return i;
133 }

176 /*FUNCTION**********************************************************************
177  *
178  * Function Name : SPI_MasterInit
179  * Description : Configures the SPI in master mode
180  * Implements : SPI_MasterInit_Activity
181  *
182  *END**************************************************************************/
183 status_t SPI_MasterInit(spi_instance_t instance, const spi_master_t *config)
184 {
185  status_t status = STATUS_ERROR;
186  uint8_t index = 0;
187 
188  /* Define SPI PAL over LPSPI */
189  #if (defined (SPI_OVER_LPSPI))
190  if ((uint32_t)instance <= LPSPI_HIGH_INDEX)
191  {
192  lpspi_master_config_t lpspiConfig;
193  lpspiConfig.bitsPerSec = config->baudRate;
194  lpspiConfig.whichPcs = (lpspi_which_pcs_t)config->ssPin;
195  lpspiConfig.pcsPolarity = (lpspi_signal_polarity_t)(!(bool)(config->ssPolarity));
196  lpspiConfig.bitcount = config->frameSize;
197  (void)CLOCK_SYS_GetFreq(g_lpspiClock[(uint32_t)instance] ,&lpspiConfig.lpspiSrcClk);
198  lpspiConfig.clkPhase = (lpspi_clock_phase_t)config->clockPhase;
199  lpspiConfig.clkPolarity = (lpspi_sck_polarity_t)config->clockPolarity;
200  lpspiConfig.lsbFirst = config->bitOrder;
201  lpspiConfig.transferType = (lpspi_transfer_type)config->transferType;
202  lpspiConfig.rxDMAChannel = config->rxDMAChannel;
203  lpspiConfig.txDMAChannel = config->txDMAChannel;
204  lpspiConfig.callback = config->callback;
205  lpspiConfig.callbackParam = config->callbackParam;
206  lpspiConfig.isPcsContinuous = false;
207  /* Allocate one of the LPSPI state structure for this instance */
208  index = SpiAllocateState(LpspiStateIsAllocated, LpspiStateInstanceMapping, instance, NO_OF_LPSPI_INSTS_FOR_SPI);
209  status = LPSPI_DRV_MasterInit(instance, (lpspi_state_t*)(&LpspiState[index]), &lpspiConfig);
210  }
211  #endif
212 
213  /* Define SPI PAL over FLEXIO */
214  #if (defined (SPI_OVER_FLEXIO))
215  if (((uint32_t)instance >= FLEXIO_SPI_LOW_INDEX) && ((uint32_t)instance <= FLEXIO_SPI_HIGH_INDEX))
216  {
217  /* FlexIO driver can be used only for the following frame sizes: 8,16 or 32 */
218  DEV_ASSERT((config->frameSize == (uint32_t)8) || (config->frameSize == (uint32_t)16) || (config->frameSize == (uint32_t)32));
219  /* FlexIO driver can be used only when SS is active low */
220  DEV_ASSERT((config->ssPolarity) == SPI_ACTIVE_LOW);
221  flexio_spi_master_user_config_t flexioConfig;
222  flexioConfig.baudRate = config->baudRate;
223  switch(config->transferType)
224  {
225  case SPI_USING_DMA: flexioConfig.driverType = FLEXIO_DRIVER_TYPE_DMA; break;
226  case SPI_USING_INTERRUPTS : flexioConfig.driverType = FLEXIO_DRIVER_TYPE_INTERRUPTS; break;
227  default : /*Nothing to do */ break;
228  }
229  flexioConfig.bitOrder = (flexio_spi_transfer_bit_order_t)config->bitOrder;
230  switch (config->frameSize)
231  {
232  case 8U : flexioConfig.transferSize = FLEXIO_SPI_TRANSFER_1BYTE; break;
233  case 16U : flexioConfig.transferSize = FLEXIO_SPI_TRANSFER_2BYTE; break;
234  case 32U : flexioConfig.transferSize = FLEXIO_SPI_TRANSFER_4BYTE; break;
235  default : /* Nothing to do */ break;
236  }
237  flexioConfig.clockPolarity = (uint8_t)config->clockPolarity;
238  flexioConfig.clockPhase = (uint8_t)config->clockPhase;
239  flexioConfig.mosiPin = ((extension_flexio_for_spi_t*)(config->extension))->mosiPin;
240  flexioConfig.misoPin = ((extension_flexio_for_spi_t*)(config->extension))->misoPin;
241  flexioConfig.sckPin = ((extension_flexio_for_spi_t*)(config->extension))->sckPin;
242  flexioConfig.ssPin = ((extension_flexio_for_spi_t*)(config->extension))->ssPin;
243  /* Link Flexio Callbacks to the callbacks defined in PAL */
244  flexioConfig.callback = config->callback;
245  /* Store the callbacks defined by the user in PAL internal callbacks vector */
246  flexioConfig.callbackParam = config->callbackParam;
247  flexioConfig.rxDMAChannel = config->rxDMAChannel;
248  flexioConfig.txDMAChannel = config->txDMAChannel;
249  status = FLEXIO_DRV_InitDevice(0U, &FlexioState);
250  if (status == STATUS_SUCCESS)
251  {
252  /* Allocate one of the Flexio state structure for this instance */
253  index = SpiAllocateState(FlexioSpiStateIsAllocated, FlexioSpiStateInstanceMapping, instance, NO_OF_FLEXIO_INSTS_FOR_SPI);
254  status = FLEXIO_SPI_DRV_MasterInit(0U, &flexioConfig, (flexio_spi_master_state_t*)&FlexioSpiState[index]);
255  }
256  }
257  #endif
258 
259  /* Define SPI PAL over DSPI */
260  #if (defined (SPI_OVER_DSPI))
261  dspi_master_config_t dspiConfig;
262  dspiConfig.bitsPerSec = config->baudRate;
263  dspiConfig.bitcount = config->frameSize;
264  dspiConfig.clkPhase = (dspi_clock_phase_t)config->clockPhase;
265  dspiConfig.clkPolarity = (dspi_polarity_t)config->clockPolarity;
266  dspiConfig.isClkContinuous = false;
267  dspiConfig.lsbFirst = config->bitOrder;
268  dspiConfig.transferType = (dspi_transfer_type_t)config->transferType;
269  dspiConfig.rxDMAChannel = config->rxDMAChannel;
270  dspiConfig.txDMAChannel = config->txDMAChannel;
271  dspiConfig.whichPCS = config->ssPin;
272  dspiConfig.pcsPolarity = (dspi_polarity_t)config->ssPolarity;
273  dspiConfig.callback = config->callback;
274  dspiConfig.callbackParam = config->callbackParam;
275  /* Allocate one of the DSPI state structure for this instance */
276  index = SpiAllocateState(DspiStateIsAllocated, DspiStateInstanceMapping, instance, NO_OF_DSPI_INSTS_FOR_SPI);
277  status = DSPI_MasterInit((dspi_instance_t)instance, (dspi_state_t*)&dspiState[index] ,&dspiConfig);
278  #endif
279  return status;
280 }
