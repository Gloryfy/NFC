172  uint8_t channel = 0U;

176  ocState = &g_ocPalStatePtr[indexInstance];
177  DEV_ASSERT(ocState->nNumChannels == 0x0U);
178 
179  /* Initialize the internal context to default value */

329  channel,

333  channel = configPtr->outputChConfig[index].hwChannelId;
334 
335  ocParam.mode = EMIOS_MODE_SAOC;

338  ocParam.matchTrailingEdgeValue = 0U;
339 
340  /* Check the operation mode of channel output */

361  &ocParam);
362 
363  /* Store some needed information into state structure */
364  ocState->channelConfigArray[index] = channel;
365  ocState->ocChannelMode[channel] = configPtr->outputChConfig[index].chMode;

368  /* Enable the global eMIOS to start counter */

375 /*FUNCTION**********************************************************************
376  *
377  * Function Name : OC_Deinit
378  * Description : This function will disable the output compare mode. The driver
379  * can't be used again until reinitialized. The context structure is no longer
380  * needed by the driver and can be freed after calling this function.
381  *
382  * Implements : OC_Deinit_Activity
383  *END**************************************************************************/
384 status_t OC_Deinit(uint32_t instance)
385 {
386  DEV_ASSERT(instance < OC_PAL_INSTANCES_MAX);
387  status_t status = STATUS_ERROR;
388  uint8_t index;
389  oc_pal_state_t * ocState;
390 
391  /* Allocate one of the OC state structure for this instance */
392  index = FindOcState(instance);
393  ocState = &g_ocPalStatePtr[index];
394 
395 #if (defined(OC_PAL_OVER_FTM))
396  /* Disable the output compare over FTM */
397  status = FTM_DRV_Deinit(instance);
398 #endif
399 
400 #if (defined(OC_PAL_OVER_EMIOS))
401  uint8_t channel;
402 
403  for (index = 0U; index < ocState->nNumChannels; index++)
404  {
405  channel = ocState->channelConfigArray[index];
406  /* Disable channels in the output compare over EMIOS */
407  EMIOS_DRV_DeInitChannel((uint8_t)instance,
408  channel);
409  }
410  status = STATUS_SUCCESS;
411 
412 #endif
413 
414  /* De-Initialize the internal context to default value */
415  ocState->nNumChannels = 0U;
416  for (index = 0U; index < OC_PAL_NUM_OF_CHANNEL_MAX; index++)
417  {
418  ocState->channelConfigArray[index] = 0U;
419  ocState->ocChannelMode[index] = OC_DISABLE_OUTPUT;
420  }
421 
422  if (status == STATUS_SUCCESS)
423  {
424  /* De-Allocate the instance which is not used */
425  ocFreeState(ocStateIsAllocated, ocStateInstanceMapping, instance, NUMBER_OF_OC_PAL_INSTANCES);
426  }
427 
428  return status;
429 }

572  *
573  * Implements : OC_SetCompareValue_Activity

575 status_t OC_SetCompareValue(uint32_t instance,
576  uint8_t channel,
577  uint32_t nextCompareMatchValue,
578  oc_option_update_t typeOfupdate)
579 {
580  DEV_ASSERT(instance < OC_PAL_INSTANCES_MAX);
581  DEV_ASSERT(channel < OC_PAL_NUM_OF_CHANNEL_MAX);
582  status_t status = STATUS_ERROR;
583 
584 #if (defined(OC_PAL_OVER_FTM))
585  /* Update the output compare value over FTM */
586  status = FTM_DRV_UpdateOutputCompareChannel(instance,
587  channel,
588  (uint16_t)nextCompareMatchValue,
589  (ftm_output_compare_update_t)typeOfupdate,
590  true);
591 #endif
592 
593 #if (defined(OC_PAL_OVER_EMIOS))
594  uint32_t compareValue = 0U;
595  uint32_t maxCounterValue = 0U;
596  uint32_t counterValue = EMIOS_DRV_MC_CounterRead((uint8_t)instance, channel);
597 
598  if (typeOfupdate == OC_RELATIVE_VALUE)
599  {
600  maxCounterValue = EMIOS_DRV_MC_GetCounterPeriod((uint8_t)instance, channel);
601  /* Configure channel compare register */
602  if (nextCompareMatchValue > (maxCounterValue - counterValue))
603  {
604  compareValue = (nextCompareMatchValue - (maxCounterValue - counterValue));
605  }
606  else
607  {
608  compareValue = (counterValue + nextCompareMatchValue);
609  }
610  }
611  else
612  {
613  compareValue = nextCompareMatchValue;
614  }
615 
616  /* Update the output compare value over EMIOS */
617  EMIOS_DRV_OC_SetSingleActOutputCmpMatch((uint8_t)instance,
618  channel,
619  compareValue);
620  status = STATUS_SUCCESS;
621 #endif
622 
623  return status;
624 }
