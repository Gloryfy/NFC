111 /*FUNCTION**********************************************************************
112  *
113  * Function Name : CAN_AllocateState
114  * Description : Allocates one of the available state structures.
115  *
116  *END**************************************************************************/
117 static uint8_t CAN_AllocateState(bool *isAllocated,
118  can_instance_t *instanceMapping,
119  can_instance_t instance,
120  uint8_t numberOfinstances)
121 {
122  uint8_t i;
123  /* Allocate one of the CAN state structures for this instance */
124  for (i = 0; i < numberOfinstances; i++)
125  {
126  if (isAllocated[i] == false)
127  {
128  instanceMapping[i] = instance;
129  isAllocated[i] = true;
130  break;
131  }
132  }
133  return i;
134 }

173 /*FUNCTION**********************************************************************
174  *
175  * Function Name : CAN_InternalCallback
176  * Description : Internal callback used to translate buffer indices
177  * and event types
178  *
179  *END**************************************************************************/
180 static void CAN_InternalCallback(uint8_t instance,
181  flexcan_event_type_t eventType,
182  uint32_t buffIdx,
183  flexcan_state_t *state)
184 {
185  can_callback_t callback = userCallbacks[instance];
186 
187  /* If FlexCAN Rx FIFO is enabled, translate real buffer index to virtual index */
188  if ((s_flexcanRxFifoState[instance].rxFifoEn == true) && (buffIdx != 0U))
189  {
190  buffIdx -= CAN_GetVirtualBuffIdx((uint32_t) s_flexcanRxFifoState[instance].numIdFilters);
191  }
192 
193  /* Translate FlexCAN events to CAN PAL events and invoke the callback provided by user */
194  switch (eventType)
195  {
196  case FLEXCAN_EVENT_TX_COMPLETE:
197  callback(instance,
198  CAN_EVENT_TX_COMPLETE,
199  (uint8_t) buffIdx,
200  (flexcan_state_t *) state);
201  break;
202  case FLEXCAN_EVENT_RX_COMPLETE:
203  callback(instance,
204  CAN_EVENT_RX_COMPLETE,
205  (uint8_t) buffIdx,
206  (flexcan_state_t *) state);
207  break;
208  case FLEXCAN_EVENT_RXFIFO_COMPLETE:
209  callback(instance,
210  CAN_EVENT_RX_COMPLETE,
211  (uint8_t) buffIdx,
212  (flexcan_state_t *) state);
213  break;
214  default:
215  /* Event types not implemented in PAL */
216  break;
217  }
218 }

233 status_t CAN_Init(can_instance_t instance, const can_user_config_t *config)
234 {
235  status_t status = STATUS_ERROR;
236  uint8_t index = 0;
237 
238  /* Define CAN PAL over FLEXCAN */
239  #if (defined (CAN_OVER_FLEXCAN))
240  if ((uint8_t)instance <= FLEXCAN_HIGH_INDEX)
241  {
242  flexcan_user_config_t flexcanConfig;
243 
244  /* Clear Rx FIFO state */
245  s_flexcanRxFifoState[instance].rxFifoEn = false;
246 
247  /* Configure features implemented by PAL */
248  flexcanConfig.max_num_mb = config->maxBuffNum;
249  flexcanConfig.flexcanMode = (flexcan_operation_modes_t) config->mode;
250 #if FEATURE_CAN_HAS_FD
251  flexcanConfig.fd_enable = config->enableFD;
252  flexcanConfig.payload = (flexcan_fd_payload_size_t) config->payloadSize;
253 #endif
254 
255  flexcanConfig.bitrate.phaseSeg1 = config->nominalBitrate.phaseSeg1;
256  flexcanConfig.bitrate.phaseSeg2 = config->nominalBitrate.phaseSeg2;
257  flexcanConfig.bitrate.preDivider = config->nominalBitrate.preDivider;
258  flexcanConfig.bitrate.propSeg = config->nominalBitrate.propSeg;
259  flexcanConfig.bitrate.rJumpwidth = config->nominalBitrate.rJumpwidth;
260 
261 #if FEATURE_CAN_HAS_FD
262  flexcanConfig.bitrate_cbt.phaseSeg1 = config->dataBitrate.phaseSeg1;
263  flexcanConfig.bitrate_cbt.phaseSeg2 = config->dataBitrate.phaseSeg2;
264  flexcanConfig.bitrate_cbt.preDivider = config->dataBitrate.preDivider;
265  flexcanConfig.bitrate_cbt.propSeg = config->dataBitrate.propSeg;
266  flexcanConfig.bitrate_cbt.rJumpwidth = config->dataBitrate.rJumpwidth;
267 #endif
268 
269 #if FEATURE_CAN_HAS_PE_CLKSRC_SELECT
270  flexcanConfig.pe_clock = (flexcan_clk_source_t) config->peClkSrc;
271 #endif
272 
273  /* If extension is used, configure Rx FIFO */
274  if (config->extension != NULL)
275  {
276  flexcanConfig.is_rx_fifo_needed = true;
277  flexcanConfig.num_id_filters = ((extension_flexcan_rx_fifo_t *)
278  (config->extension))->numIdFilters;
279 #if FEATURE_CAN_HAS_DMA_ENABLE
280  flexcanConfig.rxFifoDMAChannel = 0U;
281 #endif
282  flexcanConfig.transfer_type = FLEXCAN_RXFIFO_USING_INTERRUPTS;
283 
284  /* Compute maximum number of virtual buffers */
285  flexcanConfig.max_num_mb += CAN_GetVirtualBuffIdx(flexcanConfig.num_id_filters);
286 
287  /* Update Rx FIFO state */
288  s_flexcanRxFifoState[instance].rxFifoEn = true;
289  s_flexcanRxFifoState[instance].numIdFilters = flexcanConfig.num_id_filters;
290  }
291  else
292  {
293  flexcanConfig.is_rx_fifo_needed = false;
294  flexcanConfig.num_id_filters = FLEXCAN_RX_FIFO_ID_FILTERS_8;
295 #if FEATURE_CAN_HAS_DMA_ENABLE
296  flexcanConfig.rxFifoDMAChannel = 0U;
297 #endif
298  flexcanConfig.transfer_type = FLEXCAN_RXFIFO_USING_INTERRUPTS;
299  }
300 
301 
302  /* Allocate one of the FLEXCAN state structure for this instance */
303  index = CAN_AllocateState(s_flexcanStateIsAllocated,
304  s_flexcanStateInstanceMapping,
305  instance,
306  NO_OF_FLEXCAN_INSTS_FOR_CAN);
307  /* Initialize FLEXCAN instance */
308  status = FLEXCAN_DRV_Init((uint8_t)instance, &s_flexcanState[index], &flexcanConfig);
309 
310  /* Configure Rx FIFO if needed */
311  if ((status == STATUS_SUCCESS) && (s_flexcanRxFifoState[instance].rxFifoEn == true))
312  {
313  FLEXCAN_DRV_ConfigRxFifo(
314  (uint8_t) instance,
315  ((extension_flexcan_rx_fifo_t *) (config->extension))->idFormat,
316  ((extension_flexcan_rx_fifo_t *) (config->extension))->idFilterTable);
317  }
318  }
319  #endif
320 
321  return status;
322 }
