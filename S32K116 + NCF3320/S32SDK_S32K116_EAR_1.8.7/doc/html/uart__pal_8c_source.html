118  * Function Name : UART_AllocateState
119  * Description : Allocates one of the available state structures.
120  *
121  *END**************************************************************************/
122 static uint8_t UART_AllocateState(bool* isAllocated,
123  uart_instance_t* instanceMapping,
124  uart_instance_t instance,
125  uint8_t numberOfinstances)
126 {
127  uint8_t i;
128  /* Allocate one of the UART state structures for this instance */
129  for (i = 0; i < numberOfinstances; i++)
130  {
131  if (isAllocated[i] == false)
132  {
133  instanceMapping[i] = instance;
134  isAllocated[i] = true;
135  break;
136  }
137  }
138  return i;
139 }

191  *END**************************************************************************/
192 status_t UART_Init(uart_instance_t instance, const uart_user_config_t *config)
193 {
194  status_t status = STATUS_ERROR;
195  uint8_t index = 0;
196 
197  /* Define UART PAL over LPUART */
198  #if (defined (UART_OVER_LPUART))
199  if ((uint8_t)instance <= LPUART_HIGH_INDEX)
200  {
201  lpuart_user_config_t lpuartConfig;
202 
203  lpuartConfig.baudRate = config->baudRate;
204 
205  /* LPUART supports only 8, 9 or 10 bits per character */
206  DEV_ASSERT((config->bitCount >= UART_8_BITS_PER_CHAR) &&
207  (config->bitCount <= UART_10_BITS_PER_CHAR));
208  switch (config->bitCount)
209  {
210  case UART_8_BITS_PER_CHAR:
211  lpuartConfig.bitCountPerChar = LPUART_8_BITS_PER_CHAR;
212  break;
213  case UART_9_BITS_PER_CHAR:
214  lpuartConfig.bitCountPerChar = LPUART_9_BITS_PER_CHAR;
215  break;
216  case UART_10_BITS_PER_CHAR:
217  lpuartConfig.bitCountPerChar = LPUART_10_BITS_PER_CHAR;
218  break;
219  default:
220  /* Impossible type - do nothing */
221  break;
222  }
223 
224  lpuartConfig.parityMode = (lpuart_parity_mode_t)(config->parityMode);
225  lpuartConfig.stopBitCount = (lpuart_stop_bit_count_t)(config->stopBitCount);
226  lpuartConfig.transferType = (lpuart_transfer_type_t)(config->transferType);
227  lpuartConfig.rxDMAChannel = config->rxDMAChannel;
228  lpuartConfig.txDMAChannel = config->txDMAChannel;
229 
230  /* Allocate one of the LPUART state structure for this instance */
231  index = UART_AllocateState(s_lpuartStateIsAllocated,
232  s_lpuartStateInstanceMapping,
233  instance,
234  NO_OF_LPUART_INSTS_FOR_UART);
235  /* Initialize LPUART instance */
236  status = LPUART_DRV_Init((uint32_t)instance, (lpuart_state_t*)(&s_lpuartState[index]), &lpuartConfig);
237 
238  /* Install Rx callback */
239  if (config->rxCallback != NULL)
240  {
241  (void)LPUART_DRV_InstallRxCallback((uint32_t)instance,
242  config->rxCallback,
243  config->rxCallbackParam);
244  }
245 
246  /* Install Tx callback */
247  if (config->txCallback != NULL)
248  {
249  (void)LPUART_DRV_InstallTxCallback((uint32_t)instance,
250  config->txCallback,
251  config->txCallbackParam);
252  }
253  }
254  #endif
255 
256  /* Define UART PAL over FLEXIO */
257  #if (defined (UART_OVER_FLEXIO))
258  if (((uint8_t)instance >= FLEXIO_UART_LOW_INDEX) &&
259  ((uint8_t)instance <= FLEXIO_UART_HIGH_INDEX))
260  {
261  /* FlexIO driver can be used with parity disabled and one stop bit */
262  DEV_ASSERT(config->parityMode == UART_PARITY_DISABLED);
263  DEV_ASSERT(config->stopBitCount == UART_ONE_STOP_BIT);
264 
265  flexio_uart_user_config_t flexioUartTxConfig;
266  flexio_uart_user_config_t flexioUartRxConfig;
267 
268  /* Set baud rate for Tx and Rx */
269  flexioUartTxConfig.baudRate = config->baudRate;
270  flexioUartRxConfig.baudRate = config->baudRate;
271 
272  /* Set transfer type for Tx and Rx */
273  switch (config->transferType)
274  {
275  case UART_USING_DMA:
276  flexioUartTxConfig.driverType = FLEXIO_DRIVER_TYPE_DMA;
277  flexioUartRxConfig.driverType = FLEXIO_DRIVER_TYPE_DMA;
278  break;
279  case UART_USING_INTERRUPTS:
280  flexioUartTxConfig.driverType = FLEXIO_DRIVER_TYPE_INTERRUPTS;
281  flexioUartRxConfig.driverType = FLEXIO_DRIVER_TYPE_INTERRUPTS;
282  break;
283  default:
284  /* Impossible type - do nothing */
285  break;
286  }
287 
288  /* Set bit count per char for Tx and Rx */
289  switch (config->bitCount)
290  {
291  case UART_7_BITS_PER_CHAR:
292  flexioUartTxConfig.bitCount = 7U;
293  flexioUartRxConfig.bitCount = 7U;
294  break;
295  case UART_8_BITS_PER_CHAR:
296  flexioUartTxConfig.bitCount = 8U;
297  flexioUartRxConfig.bitCount = 8U;
298  break;
299  case UART_9_BITS_PER_CHAR:
300  flexioUartTxConfig.bitCount = 9U;
301  flexioUartRxConfig.bitCount = 9U;
302  break;
303  case UART_10_BITS_PER_CHAR:
304  flexioUartTxConfig.bitCount = 10U;
305  flexioUartRxConfig.bitCount = 10U;
306  break;
307  case UART_15_BITS_PER_CHAR:
308  flexioUartTxConfig.bitCount = 15U;
309  flexioUartRxConfig.bitCount = 15U;
310  break;
311  case UART_16_BITS_PER_CHAR:
312  flexioUartTxConfig.bitCount = 16U;
313  flexioUartRxConfig.bitCount = 16U;
314  break;
315  default:
316  /* Impossible type - do nothing */
317  break;
318  }
319 
320  /* Configure Tx */
321  flexioUartTxConfig.direction = FLEXIO_UART_DIRECTION_TX;
322  flexioUartTxConfig.dmaChannel = config->txDMAChannel;
323  flexioUartTxConfig.dataPin = ((extension_flexio_for_uart_t *)(config->extension))->dataPinTx;
324 
325  /* Configure Rx */
326  flexioUartRxConfig.direction = FLEXIO_UART_DIRECTION_RX;
327  flexioUartRxConfig.dmaChannel = config->rxDMAChannel;
328  flexioUartRxConfig.dataPin = ((extension_flexio_for_uart_t *)(config->extension))->dataPinRx;
329 
330  /* Link Flexio Callbacks to the callbacks defined in PAL */
331  flexioUartRxConfig.callback = config->rxCallback;
332  flexioUartRxConfig.callbackParam = config->rxCallbackParam;
333  flexioUartTxConfig.callback = config->txCallback;
334  flexioUartTxConfig.callbackParam = config->txCallbackParam;
335 
336  status = FLEXIO_DRV_InitDevice(0U, &flexioDeviceState);
337  if (status == STATUS_SUCCESS)
338  {
339  /* Allocate one of the Flexio state structure for this instance */
340  index = UART_AllocateState(s_flexioUartStateIsAllocated,
341  s_flexioUartStateInstanceMapping,
342  instance,
343  NO_OF_FLEXIO_INSTS_FOR_UART);
344  /* Init FlexIO UART driver for Tx */
345  status = FLEXIO_UART_DRV_Init(0U,
346  &flexioUartTxConfig,
347  (flexio_uart_state_t*)&s_flexioUartTxState[index]);
348  /* Init FlexIO UART driver for Rx */
349  status = FLEXIO_UART_DRV_Init(0U,
350  &flexioUartRxConfig,
351  (flexio_uart_state_t*)&s_flexioUartRxState[index]);
352  }
353 
354  }
355  #endif
356 
357  /* Define UART PAL over LinFlexD */
358  #if (defined (UART_OVER_LINFLEXD))
359  if (((uint8_t)instance <= LINFLEXD_UART_HIGH_INDEX))
360  {
361  linflexd_uart_user_config_t linFlexDConfig;
362 
363  linFlexDConfig.baudRate = config->baudRate;
364 
365  /* LinFlexD does not support 9 or 10 bits per character */
366  DEV_ASSERT((config->bitCount != UART_9_BITS_PER_CHAR) &&
367  (config->bitCount != UART_10_BITS_PER_CHAR));
368  switch (config->bitCount)
369  {
370  case UART_7_BITS_PER_CHAR:
371  linFlexDConfig.wordLength = LINFLEXD_UART_7_BITS;
372  break;
373  case UART_8_BITS_PER_CHAR:
374  linFlexDConfig.wordLength = LINFLEXD_UART_8_BITS;
375  break;
376  case UART_15_BITS_PER_CHAR:
377  linFlexDConfig.wordLength = LINFLEXD_UART_15_BITS;
378  break;
379  case UART_16_BITS_PER_CHAR:
380  linFlexDConfig.wordLength = LINFLEXD_UART_16_BITS;
381  break;
382  default:
383  /* Impossible type - do nothing */
384  break;
385  }
386 
387  switch (config->parityMode)
388  {
389  case UART_PARITY_DISABLED:
390  linFlexDConfig.parityCheck = false;
391  break;
392  case UART_PARITY_EVEN:
393  linFlexDConfig.parityCheck = true;
394  linFlexDConfig.parityType = LINFLEXD_UART_PARITY_EVEN;
395  break;
396  case UART_PARITY_ODD:
397  linFlexDConfig.parityCheck = true;
398  linFlexDConfig.parityType = LINFLEXD_UART_PARITY_ODD;
399  break;
400  default:
401  /* Impossible type - do nothing */
402  break;
403  }
404 
405  linFlexDConfig.stopBitsCount = (linflexd_uart_stop_bits_count_t)config->stopBitCount;
406  linFlexDConfig.transferType = (linflexd_uart_transfer_type_t)config->transferType;
407  linFlexDConfig.txDMAChannel = config->txDMAChannel;
408  linFlexDConfig.rxDMAChannel = config->rxDMAChannel;
409 
410  /* Allocate one of the LinFlexD state structures for this instance */
411  index = UART_AllocateState(s_linFlexDStateIsAllocated,
412  s_linFlexDStateInstanceMapping,
413  instance,
414  NO_OF_LINFLEXD_INSTS_FOR_UART);
415  /* Initialize LinFlexD instance */
416  status = LINFLEXD_UART_DRV_Init(instance, &s_linFlexDState[index], &linFlexDConfig);
417 
418  /* Install Rx callback */
419  if (config->rxCallback != NULL)
420  {
421  (void)LINFLEXD_UART_DRV_InstallRxCallback(instance,
422  config->rxCallback,
423  config->rxCallbackParam);
424  }
425 
426  /* Install Tx callback */
427  if (config->txCallback != NULL)
428  {
429  (void)LINFLEXD_UART_DRV_InstallTxCallback(instance,
430  config->txCallback,
431  config->txCallbackParam);
432  }
433  }
434  #endif
435  return status;
436 }

495  {
496  UART_FreeState(s_linFlexDStateIsAllocated,
497  s_linFlexDStateInstanceMapping,
498  instance,
499  NO_OF_LINFLEXD_INSTS_FOR_UART);
500  }
501  }
502  #endif
503  return status;
504 }
