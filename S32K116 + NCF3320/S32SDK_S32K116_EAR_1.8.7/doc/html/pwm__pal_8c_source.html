36 static uint8_t PwmAllocateState(bool* isAllocated, pwm_instance_t* instanceMapping, pwm_instance_t instance, uint8_t numberOfinstances)
37 {
38  uint8_t i;
39  /* Allocate one of the FTM state structure for this instance */
40  for (i = 0;i < numberOfinstances;i++)
41  {
42  if (isAllocated[i] == false)
43  {
44  instanceMapping[i] = instance;
45  isAllocated[i] = true;
46  break;
47  }
48  }
49  return i;
50 }

72 /*FUNCTION**********************************************************************
73  *
74  * Function Name : PWM_Init
75  * Description : Initialize PWM channels based on config parameter.
76  *
77  * Implements : PWM_Init_Activity
78  *END**************************************************************************/
79 status_t PWM_Init(pwm_instance_t instance, pwm_global_config_t* config)
80 {
81  /* Declare internal variables */
82  uint8_t channel;
83  uint16_t firstEdge;
84  uint8_t index;
85  /* Declare the configuration structure for FTM */
86  ftm_pwm_param_t ftmPwmConfig;
87  ftm_independent_ch_param_t pwmIndependentChannelConfig[FEATURE_FTM_CHANNEL_COUNT];
88  ftm_combined_ch_param_t pwmCombinedChannelConfig[FEATURE_FTM_CHANNEL_COUNT >> 1U];
89  ftm_user_config_t ftmGlobalConfig;
90  /* Fault control feature is not supported, but this structure is filled with values which disable fault control */
91  ftm_pwm_fault_param_t faultConfig =
92  {
93  false, /* Output pin state on fault */
94  false, /* PWM fault interrupt state */
95  0U, /* Fault filter value */
96  FTM_FAULT_CONTROL_DISABLED, /* Fault mode */
97  {
98  {
99  false, /* Fault channel state (Enabled/Disabled) */
100  false, /* Fault channel filter state (Enabled/Disabled) */
101  FTM_POLARITY_LOW, /* Fault channel state (Enabled/Disabled) */
102  },
103  {
104  false, /* Fault channel state (Enabled/Disabled) */
105  false, /* Fault channel filter state (Enabled/Disabled) */
106  FTM_POLARITY_LOW, /* Fault channel state (Enabled/Disabled) */
107  },
108  {
109  false, /* Fault channel state (Enabled/Disabled) */
110  false, /* Fault channel filter state (Enabled/Disabled) */
111  FTM_POLARITY_LOW, /* Fault channel state (Enabled/Disabled) */
112  },
113  {
114  false, /* Fault channel state (Enabled/Disabled) */
115  false, /* Fault channel filter state (Enabled/Disabled) */
116  FTM_POLARITY_LOW, /* Fault channel state (Enabled/Disabled) */
117  }
118  }
119  };
120 
121  ftmPwmConfig.pwmCombinedChannelConfig = pwmCombinedChannelConfig;
122  ftmPwmConfig.pwmIndependentChannelConfig = pwmIndependentChannelConfig;
123  ftmPwmConfig.faultConfig = &faultConfig;
124 
125  /* Because FTM has only one timebase first channel is used to configure FTM clocking*/
126  ftmGlobalConfig.ftmClockSource = ((pwm_ftm_timebase_t*)(config->pwmChannels[0].timebase))->sourceClock;
127  ftmGlobalConfig.ftmPrescaler = ((pwm_ftm_timebase_t*)(config->pwmChannels[0].timebase))->prescaler;
128  ftmPwmConfig.deadTimePrescaler = ((pwm_ftm_timebase_t*)(config->pwmChannels[0].timebase))->deadtimePrescaler;
129 
130  /* Add dummy value for frequency */
131  ftmPwmConfig.uFrequencyHZ = 0x4000;
132 
133  /* Configure FTM mode according to first channel setup.
134  * All PWM channels must be PWM_EDGE_ALIGNED/PWM_Shifted or PWM_CENTER_ALIGNED */
135  if ((config->pwmChannels[0].channelType) == PWM_CENTER_ALIGNED)
136  {
137  ftmPwmConfig.mode = FTM_MODE_CEN_ALIGNED_PWM;
138  }
139  else
140  {
141  ftmPwmConfig.mode = FTM_MODE_EDGE_ALIGNED_PWM;
142  }
143  ftmGlobalConfig.ftmMode = ftmPwmConfig.mode;
144 
145  /* The synchronization for duty, period and phase shift will be update when
146  * signal period is done. Only overwrite function shall take effect immediate.
147  */
148  ftmGlobalConfig.syncMethod.softwareSync = true;
149  ftmGlobalConfig.syncMethod.hardwareSync1 = false;
150  ftmGlobalConfig.syncMethod.hardwareSync2 = false;
151  ftmGlobalConfig.syncMethod.hardwareSync0 = false;
152  ftmGlobalConfig.syncMethod.autoClearTrigger = false;
153  ftmGlobalConfig.syncMethod.maskRegSync = FTM_SYSTEM_CLOCK;
154  ftmGlobalConfig.syncMethod.initCounterSync = FTM_PWM_SYNC;
155  ftmGlobalConfig.syncMethod.inverterSync = FTM_PWM_SYNC;
156  ftmGlobalConfig.syncMethod.outRegSync = FTM_SYSTEM_CLOCK;
157  ftmGlobalConfig.syncMethod.maxLoadingPoint = true;
158  ftmGlobalConfig.syncMethod.minLoadingPoint = false;
159  ftmGlobalConfig.syncMethod.syncPoint = FTM_WAIT_LOADING_POINTS;
160  ftmGlobalConfig.isTofIsrEnabled = false;
161  ftmGlobalConfig.BDMMode = FTM_BDM_MODE_00;
162 
163  /* Configure FTM channels */
164  ftmPwmConfig.nNumCombinedPwmChannels = 0;
165  ftmPwmConfig.nNumIndependentPwmChannels = 0;
166  for (channel = 0; channel < config->numberOfPwmChannels; channel++)
167  {
168  /* This value will be update later because initialization function doesn't support update in ticks. */
169  pwmIndependentChannelConfig[ftmPwmConfig.nNumIndependentPwmChannels].uDutyCyclePercent = 0U;
170  /* Configure channel number */
171  pwmIndependentChannelConfig[ftmPwmConfig.nNumIndependentPwmChannels].hwChannelId = config->pwmChannels[channel].channel;
172 
173  /*Configure complementary mode */
174  pwmIndependentChannelConfig[ftmPwmConfig.nNumIndependentPwmChannels].enableSecondChannelOutput = config->pwmChannels[channel].enableComplementaryChannel;
175  if ((config->pwmChannels[channel].complementaryChannelPolarity) == PWM_DUPLICATED)
176  {
177  pwmIndependentChannelConfig[ftmPwmConfig.nNumIndependentPwmChannels].secondChannelPolarity = FTM_MAIN_DUPLICATED;
178  }
179  else
180  {
181  pwmIndependentChannelConfig[ftmPwmConfig.nNumIndependentPwmChannels].secondChannelPolarity = FTM_MAIN_INVERTED;
182  }
183 
184  /* Configure channel polarity */
185  if (config->pwmChannels[channel].polarity == PWM_ACTIVE_HIGH)
186  {
187  pwmIndependentChannelConfig[ftmPwmConfig.nNumIndependentPwmChannels].polarity = FTM_POLARITY_HIGH;
188  }
189  else
190  {
191  pwmIndependentChannelConfig[ftmPwmConfig.nNumIndependentPwmChannels].polarity = FTM_POLARITY_LOW;
192  }
193  /* Configure dead-time insertion. For them the dead time configuration is available for all channels. */
194  ftmPwmConfig.deadTimePrescaler = ((pwm_ftm_timebase_t*)(config->pwmChannels[0].timebase))->deadtimePrescaler;
195  ftmPwmConfig.deadTimeValue = config->pwmChannels[0].deadtime;
196  /* Configure default value for fail safe value. */
197  pwmIndependentChannelConfig[ftmPwmConfig.nNumIndependentPwmChannels].levelSelect = FTM_HIGH_TRUE_PULSE;
198  pwmIndependentChannelConfig[ftmPwmConfig.nNumIndependentPwmChannels].enableExternalTrigger = false;
199  pwmIndependentChannelConfig[ftmPwmConfig.nNumIndependentPwmChannels].deadTime = true;
200  ftmPwmConfig.nNumIndependentPwmChannels++;
201  }
202  /* Initialize FTM as PWM generator */
203  index = PwmAllocateState(FtmStateIsAllocated, FtmStateInstanceMapping, instance, NO_OF_FTM_INSTS_FOR_PWM);
204  FTM_DRV_Init(instance, &ftmGlobalConfig, (ftm_state_t*)(&FtmState[index]));
205  FTM_DRV_InitPwm(instance, &ftmPwmConfig);
206 
207  /* Configure duty and period for all FTM PWM channels */
208  for (channel = 0; channel < config->numberOfPwmChannels; channel++)
209  {
210  DEV_ASSERT(config->pwmChannels[channel].duty <= config->pwmChannels[channel].period);
211  /* This cast is assumed because FTM counter is 16 bits wide */
212  firstEdge = (uint16_t)config->pwmChannels[channel].duty;
213  if (FtmState[index].ftmMode == FTM_MODE_CEN_ALIGNED_PWM)
214  {
215  firstEdge = firstEdge >> 1U;
216  }
217  FTM_DRV_FastUpdatePwmChannels(instance, 1U, &(config->pwmChannels[channel].channel), &firstEdge, false);
218  }
219 
220  /* Update period and generate software trigger. */
221  FTM_DRV_UpdatePwmPeriod(instance, FTM_PWM_UPDATE_IN_TICKS, config->pwmChannels[0].period , true);
222  return STATUS_SUCCESS;
223 }
