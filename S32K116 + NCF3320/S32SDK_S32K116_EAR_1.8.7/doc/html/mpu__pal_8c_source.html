87 static void MPU_ConvertMpuRegionConfig(const mpu_region_config_t * regionConfig,
88  mpu_master_access_right_t * masterAccessRight,
89  mpu_user_config_t * configResult)
90 {
91  uint8_t masterIdx = 0U;
92  mpu_master_privilege_right_t masterPrivilegeRight;
93 
94  /* Allocates region address */
95  configResult->startAddr = regionConfig->startAddr;
96  configResult->endAddr = regionConfig->endAddr;
97 
98  /* Allocates access rights for all masters */
99  for (masterIdx = 0U; masterIdx < FEATURE_MPU_MASTER_COUNT; masterIdx++)
100  {
101  masterPrivilegeRight = regionConfig->masterAccRight[masterIdx];
102 
103  /* Allocates master number */
104  masterAccessRight[masterIdx].masterNum = masterPrivilegeRight.masterNum;
105  /* Allocates access rights */
106  masterAccessRight[masterIdx].accessRight = (mpu_access_rights_t)masterPrivilegeRight.accessRight;
107  #if FEATURE_MPU_HAS_PROCESS_IDENTIFIER
108  /* Allocates process identifier */
109  masterAccessRight[masterIdx].processIdentifierEnable = ((regionConfig->processIdEnable
110  & (1UL << masterPrivilegeRight.masterNum))
111  != 0U);
112  #endif /* FEATURE_MPU_HAS_PROCESS_IDENTIFIER */
113  }
114 
115  /* Allocates master configuration in region */
116  configResult->masterAccRight = masterAccessRight;
117 
118 #if FEATURE_MPU_HAS_PROCESS_IDENTIFIER
119  /* Allocates process identifier */
120  configResult->processIdentifier = regionConfig->processIdentifier;
121  configResult->processIdMask = regionConfig->processIdMask;
122 #endif /* FEATURE_MPU_HAS_PROCESS_IDENTIFIER */
123 }

200 /*FUNCTION**********************************************************************
201  *
202  * Function Name : MPU_Init
203  * Description : Initializes memory protection unit by allocating regions
204  * and granting access rights for masters.
205  *
206  * Implements : MPU_Init_Activity
207  *END**************************************************************************/
208 status_t MPU_Init(mpu_instance_t instance,
209  uint8_t regionCnt,
210  const mpu_region_config_t * configPtr)
211 {
212  DEV_ASSERT(configPtr != NULL);
213  DEV_ASSERT(regionCnt > 0U);
214 
215  status_t status = STATUS_ERROR;
216  uint8_t regionIdx = 0U;
217 
218 #if defined(MPU_OVER_MPU)
219  DEV_ASSERT(regionCnt <= MPU_RGD_COUNT);
220 
221  mpu_user_config_t mpuConfig[MPU_RGD_COUNT];
222  mpu_master_access_right_t mpuMasterConfig[MPU_RGD_COUNT * FEATURE_MPU_MASTER_COUNT];
223 
224  /* Allocates all region configuration */
225  for (regionIdx = 0U; regionIdx < regionCnt; regionIdx++)
226  {
227  MPU_ConvertMpuRegionConfig(&configPtr[regionIdx],
228  &mpuMasterConfig[regionIdx * FEATURE_MPU_MASTER_COUNT],
229  &mpuConfig[regionIdx]);
230  }
231  /* Initializes MPU module */
232  status = MPU_DRV_Init((uint32_t)instance, regionCnt, mpuConfig);
233 
234 #elif defined(MPU_OVER_SMPU)
235  DEV_ASSERT(regionCnt <= SMPU_RGD_COUNT);
236 
237  smpu_user_config_t smpuConfig[SMPU_RGD_COUNT];
238  smpu_master_access_right_t smpuMasterConfig[SMPU_RGD_COUNT * FEATURE_SMPU_MASTER_COUNT];
239 
240  /* Allocates all region configuration */
241  for (regionIdx = 0U; regionIdx < regionCnt; regionIdx++)
242  {
243  MPU_ConvertSmpuRegionConfig(&configPtr[regionIdx],
244  &smpuMasterConfig[regionIdx * FEATURE_SMPU_MASTER_COUNT],
245  &smpuConfig[regionIdx]);
246  }
247  /* Initializes SMPU module */
248  status = SMPU_DRV_Init((uint32_t)instance, regionCnt, smpuConfig);
249 
250 #endif /* defined(MPU_OVER_MPU) & defined(MPU_OVER_SMPU) */
251 
252  return status;
253 }

281 /*FUNCTION**********************************************************************
282  *
283  * Function Name : MPU_GetDefautRegionConfig
284  * Description : Gets default region configuration. Grants all access rights for masters;
285  * disable PID and cache inhibit; unlock region descriptor.
286  * Please note that the length of master array should be defined by number of masters supported by hardware.
287  *
288  * Implements : MPU_GetDefautRegionConfig_Activity
289  *END**************************************************************************/
290 void MPU_GetDefautRegionConfig(mpu_master_privilege_right_t * masterAccRight,
291  mpu_region_config_t * regionConfig)
292 {
293  DEV_ASSERT(masterAccRight != NULL);
294  DEV_ASSERT(regionConfig != NULL);
295 
296  uint8_t masterIdx = 0U;
297 
298 #if defined(MPU_OVER_MPU)
299  uint8_t masterNum[FEATURE_MPU_MASTER_COUNT] = FEATURE_MPU_MASTER;
300 
301  /* Gets default access rights for all masters */
302  for (masterIdx = 0U; masterIdx < FEATURE_MPU_MASTER_COUNT; masterIdx++)
303  {
304  masterAccRight[masterIdx].masterNum = masterNum[masterIdx];
305  if (masterNum[masterIdx] <= FEATURE_MPU_MAX_LOW_MASTER_NUMBER)
306  {
307  masterAccRight[masterIdx].accessRight = MPU_SUPERVISOR_RWX_USER_RWX;
308  }
309  else
310  {
311  masterAccRight[masterIdx].accessRight = MPU_RW;
312  }
313  }
314 #elif defined(MPU_OVER_SMPU)
315  uint8_t masterNum[FEATURE_SMPU_MASTER_COUNT] = FEATURE_SMPU_MASTER;
316 
317  /* Gets default access rights for all masters */
318  for (masterIdx = 0U; masterIdx < FEATURE_SMPU_MASTER_COUNT; masterIdx++)
319  {
320  masterAccRight[masterIdx].masterNum = masterNum[masterIdx];
321  masterAccRight[masterIdx].accessRight = MPU_RW;
322  }
323 #endif /* defined(MPU_OVER_MPU) & defined(MPU_OVER_SMPU) */
324 
325  /* Gets default region configuration */
326  regionConfig->startAddr = 0U;
327  regionConfig->endAddr = 0xFFFFFFFFU;
328  regionConfig->masterAccRight = masterAccRight;
329  regionConfig->processIdEnable = 0U;
330  regionConfig->processIdentifier = 0U;
331  regionConfig->processIdMask = 0U;
332  regionConfig->extension = NULL;
333 }
