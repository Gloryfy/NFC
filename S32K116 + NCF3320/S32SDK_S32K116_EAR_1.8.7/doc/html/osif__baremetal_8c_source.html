102 static inline void osif_UpdateTickConfig(void)
103 {
104  uint32_t core_freq = 0u;
105  /* Get the correct name of the core clock */
106  clock_names_t coreclk = CORE_CLK;
107  status_t clk_status = CLOCK_SYS_GetFreq(coreclk, &core_freq);
108  DEV_ASSERT(clk_status == STATUS_SUCCESS);
109  DEV_ASSERT(core_freq > 0u);
110  (void)clk_status;
111 
112  /* For Cortex-M0 devices the systick counter is initialized with an undefined
113  value, so make sure to initialize it to 0 before starting */
114  S32_SysTick->CSR = S32_SysTick_CSR_ENABLE(0u);
115  S32_SysTick->RVR = S32_SysTick_RVR_RELOAD(core_freq / 1000u);
116  S32_SysTick->CVR = S32_SysTick_CVR_CURRENT(0U);
117  S32_SysTick->CSR = S32_SysTick_CSR_ENABLE(1u) | S32_SysTick_CSR_TICKINT(1u);
118 }

141 static inline void osif_UpdateTickConfig(void)
142 {
143  uint32_t tick_freq = 0u;
144  PIT_Type * base = OSIF_PIT;
145 
146  /* get the clock frequency for the timer and compute ticks for 1 ms */
147  status_t clk_status = CLOCK_SYS_GetFreq(PITRTI0_CLK, &tick_freq);
148  DEV_ASSERT(clk_status == STATUS_SUCCESS);
149  DEV_ASSERT(tick_freq > 0u);
150  (void)clk_status;
151  uint32_t tick_1ms = tick_freq / 1000u;
152 
153  /* setup timer and enable interrupt */
154  base->MCR &= ~PIT_MCR_MDIS(1u); /* make sure module is started */
155  base->MCR |= PIT_MCR_FRZ(1u); /* stop in Debug */
156  base->TIMER[OSIF_PIT_CHAN_ID].LDVAL = tick_1ms;
157  base->TIMER[OSIF_PIT_CHAN_ID].TCTRL |= PIT_TCTRL_TEN(1u) | PIT_TCTRL_TIE(1u);
158  const IRQn_Type pitIrqId[] = PIT_IRQS;
159  INT_SYS_EnableIRQ(pitIrqId[OSIF_PIT_CHAN_ID]);
160 }

196 /*FUNCTION**********************************************************************
197  *
198  * Function Name : OSIF_TimeDelay
199  * Description : This function blocks execution for a number of milliseconds.
200  *
201  * Implements : OSIF_TimeDelay_baremetal_Activity
202  *END**************************************************************************/
203 void OSIF_TimeDelay(const uint32_t delay)
204 {
205  osif_UpdateTickConfig();
206  uint32_t start = osif_GetCurrentTickCount();
207  uint32_t crt_ticks = osif_GetCurrentTickCount();
208  uint32_t delta = crt_ticks - start;
209  uint32_t delay_ticks = MSEC_TO_TICK(delay);
210  while (delta < delay_ticks)
211  {
212  crt_ticks = osif_GetCurrentTickCount();
213  delta = crt_ticks - start;
214  }
215 }

217 /*FUNCTION**********************************************************************
218  *
219  * Function Name : OSIF_GetMilliseconds
220  * Description : This function returns the number of miliseconds elapsed since
221  * starting the internal timer. To initialize the internal timer
222  * (Systick) in bare-metal, call either OSIF_TimeDelay or
223  * OSIF_SemaWait functions. Calling OSIF_TimeDelay(0) will initialize
224  * the timer without any side-effects (no delay).
225  *
226  * Implements : OSIF_GetMilliseconds_baremetal_Activity
227  *END**************************************************************************/
228 uint32_t OSIF_GetMilliseconds(void)
229 {
230  /*
231  * Please make sure the timer is initialized before calling this function.
232  * For example, calling OSIF_TimeDelay(0) ensures that the timer is initialized
233  * without any other side-effects. If OSIF_TimeDelay or OSIF_SemaWait functions
234  * have been called, the timer is already initialized.
235  */
236  return osif_GetCurrentTickCount(); /* This assumes that 1 tick = 1 millisecond */
237 }

297 /*FUNCTION**********************************************************************
298  *
299  * Function Name : OSIF_SemaWait
300  * Description : This function performs the 'wait' (decrement) operation on a semaphore.
301  *
302  * Implements : OSIF_SemaWait_baremetal_Activity
303  *END**************************************************************************/
304 status_t OSIF_SemaWait(semaphore_t * const pSem,
305  const uint32_t timeout)
306 {
307  DEV_ASSERT(pSem != NULL);
308 
309  uint32_t timeoutTicks;
310  status_t osif_ret_code = STATUS_SUCCESS;
311 
312  osif_UpdateTickConfig();
313  /* Convert timeout from milliseconds to ticks. */
314  if (timeout == OSIF_WAIT_FOREVER)
315  {
316  timeoutTicks = OSIF_WAIT_FOREVER;
317  }
318  else
319  {
320  timeoutTicks = MSEC_TO_TICK(timeout);
321  }
322 
323  uint32_t start = osif_GetCurrentTickCount();
324  uint32_t end = (uint32_t)(start + timeoutTicks);
325  uint32_t max = end - start;
326  while (*pSem == 0u)
327  {
328  uint32_t crt_ticks = osif_GetCurrentTickCount();
329  uint32_t delta = crt_ticks - start;
330  if ((timeoutTicks != OSIF_WAIT_FOREVER) && (delta > max))
331  {
332  /* Timeout occured, stop waiting and return fail code */
333  osif_ret_code = STATUS_TIMEOUT;
334  break;
335  }
336  }
337 
338  if (osif_ret_code == STATUS_SUCCESS)
339  {
340  osif_DisableIrqGlobal();
341  --(*pSem);
342  osif_EnableIrqGlobal();
343  }
344 
345  return osif_ret_code;
346 }
