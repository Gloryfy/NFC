103 /*FUNCTION**********************************************************************
104  *
105  * Function Name : ADC_Init
106  * Description : This function initializes the ADC PAL instance, including the
107  * other platform specific HW units used together with ADC.
108  *
109  * Implements : ADC_Init_Activity
110  *END**************************************************************************/
111 status_t ADC_Init(const adc_pal_instance_t instance, const adc_config_t * const config)
112 {
113  DEV_ASSERT(instance < NUMBER_OF_ADC_PAL_INSTANCES);
114  DEV_ASSERT(config != NULL);
115  DEV_ASSERT(config->extension != NULL);
116  DEV_ASSERT(config->groupConfigArray != NULL);
117 
118  status_t status = STATUS_SUCCESS;
119  uint16_t idx = 0;
120  (void) idx;
121 
122  /* Initialize PAL state structure */
123  adcPalState[instance].groupArray = config->groupConfigArray;
124  adcPalState[instance].activeGroupIdx = 0u;
125  adcPalState[instance].numGroups = config->numGroups;
126  adcPalState[instance].activeGroupFlag = false;
127 
128 #if defined(ADC_PAL_S32K1xx)
129  extension_adc_s32k1xx_t * extension = (extension_adc_s32k1xx_t *)(config->extension);
130  adc_converter_config_t adcCfg;
131 
132  DEV_ASSERT(extension->inputClock <= NUMBER_OF_ALT_CLOCKS);
133 
134 #if defined (CUSTOM_DEVASSERT) || defined (DEV_ERROR_DETECT)
135  {
136  /* Iterate over all conversion groups and DEV_ASSERT if number of conversions is larger than max supported */
137  for(idx = 0u; idx < adcPalState[instance].numGroups; idx++)
138  {
139  DEV_ASSERT(adcPalState[instance].groupArray[idx].numChannels <= ADC_PAL_MAX_CONVS_IN_GROUP);
140  }
141  }
142 #endif /* defined (CUSTOM_DEVASSERT) || defined (DEV_ERROR_DETECT) */
143 
144  /* ADC configuration */
145  ADC_DRV_Reset(instance);
146 
147  ADC_DRV_InitConverterStruct(&adcCfg);
148  adcCfg.trigger = ADC_TRIGGER_HARDWARE; /* ADC in hw trigger mode - sw triggering shall be done via PDB */
149  adcCfg.sampleTime = config->sampleTicks;
150  adcCfg.clockDivide = extension->clockDivide;
151  adcCfg.resolution = extension->resolution;
152  adcCfg.inputClock = extension->inputClock;
153  adcCfg.voltageRef = extension->voltageRef;
154  adcCfg.supplyMonitoringEnable = extension->supplyMonitoringEnable;
155  adcCfg.pretriggerSel = ADC_PRETRIGGER_SEL_PDB; /* configure pretriggers 0->3 to be routed from PDB */
156 
157  ADC_DRV_ConfigConverter(instance, &adcCfg);
158 
159  ADC_DRV_AutoCalibration(instance);
160 
161  /* PDB init shall only be called from StartConversion() & EnableHardwareTrigger()
162  * because PDB input source and continuous conversion enable need to be configured for each call. */
163 
164  /* Only reset the trgmux target register corresponding to PDB instance.
165  * Calling TRGMUX init would reset all TRGMUX target registers - affecting other modules. */
166  status = TRGMUX_DRV_SetTrigSourceForTargetModule(ADC_PAL_TRGMUX_IDX, TRGMUX_TRIG_SOURCE_DISABLED, adcPalTrgmuxTarget[instance]);
167  if(status != STATUS_SUCCESS)
168  {
169  status = STATUS_ERROR;
170  }
171 
172 #elif defined(ADC_PAL_MPC574xC_G_R)
173 
174  const extension_adc_mpc574xc_g_r_t * const extension = (extension_adc_mpc574xc_g_r_t *)(config->extension);
175  const adc_group_config_t * currentGroup;
176  adc_conv_config_t adcCfg;
177  bctu_config_t bctuCfg;
178  status_t bctuStatus;
179 
180  ADC_DRV_GetDefaultConfigConverter(&adcCfg);
181  adcCfg.sampleTimeExternal = config->sampleTicks;
182  adcCfg.sampleTimeInternal = config->sampleTicks;
183  adcCfg.sampleTimePrecision = config->sampleTicks;
184  adcCfg.clkSelect = extension->clkSelect;
185  adcCfg.refSelect = extension->refSelect;
186  adcCfg.autoClockOff = extension->autoClockOff;
187  adcCfg.convMode = ADC_CONV_MODE_ONESHOT;
188  adcCfg.ctuMode = ADC_CTU_MODE_TRIGGER; /* Enable ADC triggering from BCTU and also normal chains - used for SW triggered conversion groups */
189  adcCfg.overwriteEnable = true;
190 
191  /* Configure only the ADC instances which are used,
192  * i.e. which have at least one channel assigned to at least one group. */
193  uint16_t activeAdcMask = ADC_GetUsedAdcInstances(instance);
194  for(idx = 0u; idx < FEATURE_BCTU_NUM_ADC; idx++)
195  {
196  if((activeAdcMask & (1u << idx)) != 0u)
197  {
198  ADC_DRV_Reset(idx);
199  ADC_DRV_DoCalibration(idx);
200 
201  ADC_DRV_ConfigConverter(idx, &adcCfg);
202 
203  /* Used for SW triggered groups */
204  ADC_DRV_EnableInterrupts(idx, ADC_FLAG_NORMAL_ENDCHAIN);
205  }
206  }
207 
208  /* Configure BCTU */
209  bctuStatus = BCTU_DRV_Reset(instance, /* default timeout value 2ms*/ 2u);
210  if(bctuStatus == STATUS_TIMEOUT)
211  {
212  status = STATUS_BUSY;
213  }
214  else
215  {
216  BCTU_DRV_GetDefaultConfig(&bctuCfg);
217  bctuCfg.lowPowerModeEn = true; /* Low power mode is enabled by default until a HW triggered group is enabled (only HW triggered groups use BCTU) */
218  /* Used for HW triggered groups with a single conversion. By default enable new data interrupt for both ADCs */
219  bctuCfg.newDataIntEnMask = (((uint8_t)1u << FEATURE_BCTU_NUM_ADC) - 1u);
220  bctuCfg.listIntEn = true; /* Used for HW triggered groups with multiple conversions */
221 
222  BCTU_DRV_Config(instance, &bctuCfg);
223 
224  uint8_t bctuNumChansList = 0u; /* total number of channels programmed in BCTU list */
225 
226  /* Configure all HW triggered groups in BCTU.
227  * SW triggered groups will be configured directly in ADC, each time ADC_StartGroupConversion() is called. */
228  for(idx = 0u; idx < config->numGroups; idx++)
229  {
230  currentGroup = &(config->groupConfigArray[idx]);
231  DEV_ASSERT(currentGroup->inputChannelArray != NULL);
232 
233  if(currentGroup->hwTriggerSupport == true)
234  {
235  status = ADC_ConfigHwTriggeredGroup(instance, currentGroup, bctuNumChansList);
236 
237  if(currentGroup->numChannels > 1u)
238  {
239  bctuNumChansList += currentGroup->numChannels; /* Increment total number of channels programmed into BCTU LIST */
240  }
241  }
242  }
243 
244 #if defined (CUSTOM_DEVASSERT) || defined (DEV_ERROR_DETECT)
245  {
246  /* Check that a channel doesn't appear multiple times inside a group */
247  for(idx = 0u; idx < config->numGroups; idx++)
248  {
249  uint8_t numOccurrences[ADC_CDR_COUNT]; /* total number of ADC channels is given by ADC_CDR_COUNT */
250  uint8_t idx1;
251  uint8_t chanIdx;
252  currentGroup = &(config->groupConfigArray[idx]);
253 
254  /* First reset the number of occurrences of each channel */
255  for(idx1 = 0u; idx1 < currentGroup->numChannels; idx1++)
256  {
257  chanIdx = ADC_GetChanIdx(currentGroup->inputChannelArray[idx1]);
258  numOccurrences[chanIdx] = 0u;
259  }
260  /* Count the number of occurrences for each channel */
261  for(idx1 = 0u; idx1 < currentGroup->numChannels; idx1++)
262  {
263  chanIdx = ADC_GetChanIdx(currentGroup->inputChannelArray[idx1]);
264  numOccurrences[chanIdx]++;
265  }
266  for(idx1 = 0u; idx1 < currentGroup->numChannels; idx1++)
267  {
268  chanIdx = ADC_GetChanIdx(currentGroup->inputChannelArray[idx1]);
269  if(numOccurrences[chanIdx] > 1u)
270  {
271  DEV_ASSERT(false); /* A channel must not appear multiple times inside a group */
272  }
273  }
274  }
275  }
276 #endif /* defined (CUSTOM_DEVASSERT) || defined (DEV_ERROR_DETECT) */
277 
278 
279  /* Move BCTU out of configuration mode */
280  BCTU_DRV_WriteGlobalTriggerEn(instance, true);
281  }
282 #endif /* defined(ADC_PAL_MPC574xC_G_R) */
283 
284  return status;
285 }

288 /*FUNCTION**********************************************************************
289  *
290  * Function Name : ADC_Deinit
291  * Description : This function resets the ADC PAL instance, including
292  * the other platform specific HW units used together with ADC.
293  *
294  * Implements : ADC_Deinit_Activity
295  *END**************************************************************************/
296 status_t ADC_Deinit(const adc_pal_instance_t instance)
297 {
298  DEV_ASSERT(instance < NUMBER_OF_ADC_PAL_INSTANCES);
299 
300  adc_pal_state_t * const palState = &(adcPalState[instance]);
301  adc_group_state_t * const groupState = &(palState->activeGroupState);
302  status_t status;
303 
304  if(palState->activeGroupFlag == true)
305  {
306  status = STATUS_BUSY;
307  }
308  else
309  {
310  /* Reset PAL state structure */
311  palState->groupArray = NULL;
312  palState->numGroups = 0u;
313  groupState->bufferLength = 0u;
314  groupState->currentBufferOffset = 0u;
315  groupState->notificationEn = false;
316 
317 #if defined(ADC_PAL_S32K1xx)
318 
319  PDB_DRV_Deinit(instance);
320 
321  status = TRGMUX_DRV_SetTrigSourceForTargetModule(ADC_PAL_TRGMUX_IDX, TRGMUX_TRIG_SOURCE_DISABLED, adcPalTrgmuxTarget[instance]);
322  DEV_ASSERT(status == STATUS_SUCCESS);
323 
324  ADC_DRV_Reset(instance);
325 
326  /* Disable interrupt from INT manager */
327  IRQn_Type adcIrqId;
328  adcIrqId = ADC_DRV_GetInterruptNumber(instance);
329  INT_SYS_DisableIRQ(adcIrqId);
330 
331  status = STATUS_SUCCESS;
332 
333 #elif defined(ADC_PAL_MPC574xC_G_R)
334 
335  /* Deinit BCTU and ADCs */
336  status_t bctuStatus;
337  uint8_t idx;
338  bctuStatus = BCTU_DRV_Reset(instance, /* default timeout value 2ms*/ 2u);
339  if(bctuStatus == STATUS_TIMEOUT)
340  {
341  status = STATUS_BUSY;
342  }
343  else
344  {
345  /* Deinit only the active ADC instances,
346  * i.e. which have at least one channel assigned to at least one group. */
347  uint16_t activeAdcMask = ADC_GetUsedAdcInstances(instance);
348  for(idx = 0u; idx < FEATURE_BCTU_NUM_ADC; idx++)
349  {
350  if((activeAdcMask & (1u << idx)) != 0u)
351  {
352  ADC_DRV_Reset(idx);
353  }
354  }
355 
356  status = STATUS_SUCCESS;
357  }
358 #endif /* defined(ADC_PAL_MPC574xC_G_R) */
359  }
360 
361  return status;
362 }

364 /*FUNCTION**********************************************************************
365  *
366  * Function Name : ADC_EnableHardwareTrigger
367  * Description : Enables the selected HW trigger for a conversion group,
368  * if the conversion group has support for HW trigger.
369  *
370  * Implements : ADC_EnableHardwareTrigger_Activity
371  *END**************************************************************************/
372 status_t ADC_EnableHardwareTrigger(const adc_pal_instance_t instance, const uint32_t groupIdx)
373 {
374  DEV_ASSERT(instance < NUMBER_OF_ADC_PAL_INSTANCES);
375  DEV_ASSERT(groupIdx < adcPalState[instance].numGroups);
376  DEV_ASSERT(adcPalState[instance].groupArray != NULL);
377 
378  const adc_pal_state_t * const palState = &(adcPalState[instance]);
379  const adc_group_config_t * const currentGroupCfg = &(palState->groupArray[groupIdx]);
380  status_t status = STATUS_SUCCESS;
381  status_t statusTrgmux = STATUS_SUCCESS;
382  (void) statusTrgmux;
383 
384  DEV_ASSERT(currentGroupCfg->hwTriggerSupport == true);
385 
386  if(palState->activeGroupFlag == true)
387  {
388  /* A conversion group is already active */
389  status = STATUS_BUSY;
390  }
391  else
392  {
393  /* Update ADC PAL and group state structures */
394  ADC_PalStateUpdateStart(instance, groupIdx);
395 
396 #if defined(ADC_PAL_S32K1xx)
397 
398  /* Enable the ADC interrupt from Interrupt Manager */
399  IRQn_Type adcIrqId;
400  adcIrqId = ADC_DRV_GetInterruptNumber(instance);
401  INT_SYS_EnableIRQ(adcIrqId);
402 
403  /* The group shall be configured each time it is enabled. */
404  ADC_ConfigGroup(instance, groupIdx, true);
405 
406  /* Enable in TRGMUX the selected HW trigger source for PDB.
407  * Must be called after all PDB pre-triggers have been configured, to make sure no triggers occur during configuration. */
408  statusTrgmux = TRGMUX_DRV_SetTrigSourceForTargetModule(ADC_PAL_TRGMUX_IDX, currentGroupCfg->triggerSource, adcPalTrgmuxTarget[instance]);
409  DEV_ASSERT(statusTrgmux == STATUS_SUCCESS);
410 
411 #elif defined(ADC_PAL_MPC574xC_G_R)
412  const uint8_t numChans = (palState->groupArray[palState->activeGroupIdx]).numChannels;
413 
414  if(numChans == 1u)
415  {
416  INT_SYS_EnableIRQ(BCTU_ConvUpdate_IRQn);
417  }
418  else
419  {
420  INT_SYS_EnableIRQ(BCTU_ListLast_IRQn);
421  }
422 
423  /* De-activate low-power mode. Any pending triggers will become active, immediately triggering a conversion. */
424  BCTU_DRV_SetLowPowerMode(instance, false);
425 
426  /* All groups get configured at ADC PAL initialization,
427  * so here they only need to be enabled. */
428  BCTU_DRV_EnableHwTrigger(instance, currentGroupCfg->triggerSource);
429 
430 #endif /* defined(ADC_PAL_MPC574xC_G_R) */
431 
432  status = STATUS_SUCCESS;
433  }
434 
435  return status;
436 }

439 /*FUNCTION**********************************************************************
440  *
441  * Function Name : ADC_DisableHardwareTrigger
442  * Description : Disables the selected HW trigger for a conversion group,
443  * if the conversion group has support for HW trigger.
444  *
445  * Implements : ADC_DisableHardwareTrigger_Activity
446  *END**************************************************************************/
447 status_t ADC_DisableHardwareTrigger(const adc_pal_instance_t instance, const uint32_t groupIdx, const uint32_t timeout)
448 {
449  DEV_ASSERT(instance < NUMBER_OF_ADC_PAL_INSTANCES);
450  DEV_ASSERT(groupIdx < adcPalState[instance].numGroups);
451  DEV_ASSERT(adcPalState[instance].groupArray != NULL);
452  DEV_ASSERT(adcPalState[instance].groupArray[groupIdx].hwTriggerSupport == true);
453 
454  status_t status = STATUS_SUCCESS;
455 
456  if(adcPalState[instance].activeGroupIdx == groupIdx)
457  {
458 #if defined(ADC_PAL_S32K1xx)
459  /* Disable from TRGMUX the trigger source for PDB */
460  status = TRGMUX_DRV_SetTrigSourceForTargetModule(ADC_PAL_TRGMUX_IDX, TRGMUX_TRIG_SOURCE_DISABLED, adcPalTrgmuxTarget[instance]);
461  DEV_ASSERT(status == STATUS_SUCCESS);
462 
463  status = ADC_StopGroupBlocking(instance, timeout);
464 
465 #elif defined(ADC_PAL_MPC574xC_G_R)
466  const adc_pal_state_t * const palState = &(adcPalState[instance]);
467  const adc_trigger_source_t triggerSource = palState->groupArray[groupIdx].triggerSource;
468  const uint8_t numChans = (palState->groupArray[palState->activeGroupIdx]).numChannels;
469  uint32_t startTime, deltaTime;
470 
471  if(numChans == 1u)
472  {
473  INT_SYS_DisableIRQ(BCTU_ConvUpdate_IRQn);
474  }
475  else
476  {
477  INT_SYS_DisableIRQ(BCTU_ListLast_IRQn);
478  }
479 
480  OSIF_TimeDelay(0u); /* Make sure OSIF timer is initialized. */
481 
482  startTime = OSIF_GetMilliseconds();
483  deltaTime = 0u;
484 
485  /* Prevent new conversions from starting */
486  BCTU_DRV_DisableHwTrigger(instance, triggerSource);
487 
488  /* Check if conversion is still running until execution finished or timeout occurred */
489  while((BCTU_DRV_IsConvRunning(instance, triggerSource) == true) && (deltaTime < timeout))
490  {
491  deltaTime = OSIF_GetMilliseconds() - startTime;
492  }
493 
494  if(deltaTime >= timeout)
495  {
496  status = STATUS_TIMEOUT;
497  }
498  else
499  {
500  status = STATUS_SUCCESS;
501 
502  /* Activate low-power mode */
503  BCTU_DRV_SetLowPowerMode(instance, true);
504  }
505 
506 #endif /* defined(ADC_PAL_MPC574xC_G_R) */
507 
508  if(status == STATUS_SUCCESS)
509  {
510  adcPalState[instance].activeGroupFlag = false;
511  }
512  }
513 
514  return status;
515 }

517 /*FUNCTION**********************************************************************
518  *
519  * Function Name : ADC_StartGroupConversion
520  * Description : Starts the execution of a selected ADC conversion group.
521  *
522  * Implements : ADC_StartGroupConversion_Activity
523  *END**************************************************************************/
524 status_t ADC_StartGroupConversion(const adc_pal_instance_t instance, const uint32_t groupIdx)
525 {
526  DEV_ASSERT(instance < NUMBER_OF_ADC_PAL_INSTANCES);
527 
528  const adc_pal_state_t * const palState = &(adcPalState[instance]);
529 
530  DEV_ASSERT(groupIdx < palState->numGroups);
531  DEV_ASSERT(palState->groupArray != NULL);
532  DEV_ASSERT(palState->groupArray[groupIdx].hwTriggerSupport == false);
533 
534  status_t status;
535 
536  if(palState->activeGroupFlag == true)
537  {
538  /* A conversion group is already active */
539  status = STATUS_BUSY;
540  }
541  else
542  {
543  /* Update ADC PAL and group state structures */
544  ADC_PalStateUpdateStart(instance, groupIdx);
545 
546 #if defined(ADC_PAL_S32K1xx)
547  DEV_ASSERT(palState->groupArray[groupIdx].numChannels <= ADC_PAL_MAX_CONVS_IN_GROUP);
548 
549  bool hwTriggerEnabled = false;
550 
551  /* Enable the ADC interrupt from Interrupt Manager */
552  IRQn_Type adcIrqId;
553  adcIrqId = ADC_DRV_GetInterruptNumber(instance);
554  INT_SYS_EnableIRQ(adcIrqId);
555 
556  ADC_ConfigGroup(instance, groupIdx, hwTriggerEnabled);
557 
558  /* Sw trigger PDB */
559  PDB_DRV_SoftTriggerCmd(instance);
560 
561 #elif defined(ADC_PAL_MPC574xC_G_R)
562 
563  const adc_group_config_t * currentGroupCfg = &(palState->groupArray[groupIdx]);
564  uint8_t idx, adcIdx;
565  uint32_t currentChan;
566 
567  /* Assumes that all channels in the group target the same ADC instance */
568  adcIdx = ADC_GetAdcIdx(currentGroupCfg->inputChannelArray[0u]);
569 
570  switch(adcIdx)
571  {
572  case 0u:
573  INT_SYS_EnableIRQ(ADC0_EOC_IRQn);
574  break;
575  case 1u:
576  INT_SYS_EnableIRQ(ADC1_EOC_IRQn);
577  break;
578  default:
579  DEV_ASSERT(false);
580  break;
581  }
582 
583  for(idx = 0u; idx < currentGroupCfg->numChannels; idx++)
584  {
585  DEV_ASSERT(adcIdx == ADC_GetAdcIdx(currentGroupCfg->inputChannelArray[idx])); /* channels in the group must target the same ADC instance */
586  currentChan = ADC_GetChanIdx(currentGroupCfg->inputChannelArray[idx]);
587 
588  ADC_DRV_EnableChannel(adcIdx, ADC_CONV_CHAIN_NORMAL, currentChan);
589  }
590 
591  ADC_DRV_StartConversion(adcIdx, ADC_CONV_CHAIN_NORMAL);
592 
593 #endif /* defined(ADC_PAL_MPC574xC_G_R) */
594 
595  status = STATUS_SUCCESS;
596  }
597 
598  return status;
599 }

601 /*FUNCTION**********************************************************************
602  *
603  * Function Name : ADC_StopGroupConversion
604  * Description : Stops the execution of the ADC group currently executing.
605  *
606  * Implements : ADC_StopGroupConversion_Activity
607  *END**************************************************************************/
608 status_t ADC_StopGroupConversion(const adc_pal_instance_t instance, const uint32_t groupIdx, const uint32_t timeout)
609 {
610  DEV_ASSERT(instance < NUMBER_OF_ADC_PAL_INSTANCES);
611  DEV_ASSERT(groupIdx < adcPalState[instance].numGroups);
612  DEV_ASSERT(adcPalState[instance].groupArray != NULL);
613  DEV_ASSERT(adcPalState[instance].groupArray[groupIdx].hwTriggerSupport == false);
614 
615  status_t status = STATUS_SUCCESS;
616 
617  if((adcPalState[instance].activeGroupIdx == groupIdx) && (adcPalState[instance].activeGroupFlag == true))
618  {
