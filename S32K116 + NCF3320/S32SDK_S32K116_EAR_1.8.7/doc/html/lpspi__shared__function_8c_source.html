126 void LPSPI_DRV_FillupTxBuffer(uint32_t instance)
127 {
128  /* Instantiate local variable of type dspi_master_state_t and point to global state. */
129  lpspi_state_t * lpspiState = g_lpspiStatePtr[instance];
130  LPSPI_Type *base = g_lpspiBase[instance];
131  uint32_t wordToSend = 0;
132  uint16_t numOfBytes;
133  uint16_t index;
134  uint8_t availableSpace = (uint8_t)(lpspiState->fifoSize - (uint8_t)LPSPI_ReadTxCount(base));
135 
136  /* Fill the TX buffer. */
137  while(availableSpace != 0U)
138  {
139  if (lpspiState->isPcsContinuous == true)
140  {
141  if(lpspiState->txCount == 1U)
142  {
143  /* Disable continuous PCS */
144  LPSPI_ClearContCBit(base);
145  lpspiState->txCount = 0U;
146  break;
147  }
148  }
149  /* Get the number of bytes which can be written in a single 32 bits word. */
150  if ((lpspiState->bytesPerFrame - lpspiState->txFrameCnt) <= (uint16_t)4)
151  {
152  numOfBytes = (uint16_t)(lpspiState->bytesPerFrame - lpspiState->txFrameCnt);
153  }
154  else
155  {
156  numOfBytes = 4U;
157  }
158  wordToSend = 0;
159 
160  if (lpspiState->txBuff != NULL)
161  {
162  /* For the case when frame size > 4 bytes and MSB the words must be written in
163  * specific order because the hardware MSB/LSB is not enough.
164  */
165  if ((lpspiState->lsb == false) && (lpspiState->bytesPerFrame > 4U))
166  {
167  index = (uint16_t)(lpspiState->bytesPerFrame - 1U - lpspiState->txFrameCnt);
168  if(index >= 3U)
169  {
170  wordToSend = *(const uint32_t *)(lpspiState->txBuff + index - 3U);
171  }
172  else
173  {
174  wordToSend = *(const uint32_t *)(lpspiState->txBuff);
175  }
176  lpspiState->txFrameCnt = (uint16_t)((lpspiState->txFrameCnt + numOfBytes) % lpspiState->bytesPerFrame);
177  if (lpspiState->txFrameCnt == 0U)
178  {
179  lpspiState->txBuff = lpspiState->txBuff + lpspiState->bytesPerFrame;
180  }
181  }
182  else
183  {
184  switch(lpspiState->bytesPerFrame)
185  {
186  case 1:
187  wordToSend = *((const uint8_t *)(lpspiState->txBuff));
188  break;
189  case 2:
190  wordToSend = *((const uint16_t *)(lpspiState->txBuff));
191  break;
192  default:
193  wordToSend = *((const uint32_t *)(lpspiState->txBuff));
194  break;
195  }
196  lpspiState->txBuff += numOfBytes;
197  lpspiState->txFrameCnt = (uint16_t)((lpspiState->txFrameCnt + numOfBytes) % lpspiState->bytesPerFrame);
198  }
199  }
200  LPSPI_WriteData(base, wordToSend);
201  /* Update internal variable used in transmission. */
202  lpspiState->txCount = (uint16_t)(lpspiState->txCount - numOfBytes);
203  /* Verify if all bytes were send. */
204  if (lpspiState->txCount == 0U)
205  {
206  break;
207  }
208  availableSpace = (uint8_t)(availableSpace - 1U);
209  }
210 }
217 void LPSPI_DRV_ReadRXBuffer(uint32_t instance)
218 {
219  lpspi_state_t * lpspiState = g_lpspiStatePtr[instance];
220  const LPSPI_Type *base = g_lpspiBase[instance];
221  uint32_t receivedWord;
222  uint16_t numOfBytes;
223  uint16_t j;
224  uint16_t index;
225  uint8_t filledSpace = (uint8_t)LPSPI_ReadRxCount(base);
226  while (filledSpace!= 0U)
227  {
228  receivedWord = LPSPI_ReadData(base);
229  /* Get the number of bytes which can be read from this 32 bites */
230  if ((lpspiState->bytesPerFrame - lpspiState->rxFrameCnt) <= (uint16_t)4)
231  {
232  numOfBytes = (uint16_t)(lpspiState->bytesPerFrame - lpspiState->rxFrameCnt);
233  }
234  else
235  {
236  numOfBytes = 4U;
237  }
238  /* Generate the word which will be write in buffer. */
239  if ((lpspiState->lsb == false) && (lpspiState->bytesPerFrame > 4U))
240  {
241  index = (uint16_t)(lpspiState->bytesPerFrame - lpspiState->rxFrameCnt - 1U);
242  for (j = numOfBytes; j > 0U; j--)
243  {
244  lpspiState->rxBuff[index] = (uint8_t)(receivedWord >> ((j - 1U) * 8U));
245  index--;
246  }
247  lpspiState->rxFrameCnt = (uint16_t)((lpspiState->rxFrameCnt + numOfBytes) % lpspiState->bytesPerFrame);
248  if (lpspiState->rxFrameCnt == 0U)
249  {
250  lpspiState->rxBuff = lpspiState->rxBuff + lpspiState->bytesPerFrame;
251  }
252  }
253  else
254  {
255  for (j = 0; j < numOfBytes; j++)
256  {
257  *(lpspiState->rxBuff) = (uint8_t)(receivedWord >> (j * 8U));
258  lpspiState->rxBuff++;
259  }
260  lpspiState->rxFrameCnt = (uint16_t)((lpspiState->rxFrameCnt + numOfBytes) % lpspiState->bytesPerFrame);
261  }
262 
263  /* Update internal variable used in transmission. */
264  lpspiState->rxCount = (uint16_t)(lpspiState->rxCount - numOfBytes);
265  /* Verify if all bytes were sent. */
266  if (lpspiState->rxCount == 0U)
267  {
268  break;
269  }
270  filledSpace = (uint8_t)(filledSpace - 1U);
271  }
272 }
